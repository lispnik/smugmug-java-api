015    package com.kallasoft.smugmug.api;
016    
017    import org.apache.commons.httpclient.HttpClient;
018    import org.apache.commons.httpclient.MultiThreadedHttpConnectionManager;
019    
020    /* TODO: Remove the APIKey argument from every method except the login methods */
021    
022    /**
023     * Class used to provide global constants across the entire API.
024     * <p>
025     * This is defined as a <code>class</code> and not an <code>interface</code>
026     * in order to support the static initializer used to init the
027     * {@link #HTTP_CLIENT} instance.
028     *
029     * @author Riyad Kalla
030     */
031    public class APIConstants {
032            /**
033             * Used to store the current version of the API, for use in other portions
034             * of the API like reporting the User-Agent header.
035             */
036            public static final String VERSION = "0.6.0";
037    
038            /**
039             * Used to define the String used for the User-Agent HTTP header reported by
040             * the API when it connects to SmugMug.
041             */
042            public static final String USER_AGENT = "kallasoft API/" + VERSION;
043    
044            /**
045             * The singleton reference to the {@link HttpClient} used by the kallasoft
046             * SmugMug Java API to communicate with the SmugMug server. Having one
047             * instance per application is the preferred method.
048             * <p>
049             * This instance of <code>HttpClient</code> is configured specifically for
050             * safe, optimized, multi-threaded use by a static initializer in this
051             * class.
052             *
053             * @see <a
054             *      href="http://jakarta.apache.org/httpcomponents/httpclient-3.x/performance.html">HttpClient
055             *      Performance Guide</a>
056             */
057            public static final HttpClient HTTP_CLIENT = new HttpClient();
058    
059            static {
060                    MultiThreadedHttpConnectionManager connectionManager = new MultiThreadedHttpConnectionManager();
061                    HTTP_CLIENT.setHttpConnectionManager(connectionManager);
062            }
063    }
