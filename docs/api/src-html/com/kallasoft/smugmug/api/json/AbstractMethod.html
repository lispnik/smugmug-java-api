015    package com.kallasoft.smugmug.api.json;
016    
017    import org.apache.commons.httpclient.HttpStatus;
018    import org.apache.commons.httpclient.methods.PostMethod;
019    import org.apache.commons.io.IOUtils;
020    import org.slf4j.Logger;
021    import org.slf4j.LoggerFactory;
022    
023    import com.kallasoft.smugmug.api.APIConstants;
024    import com.kallasoft.smugmug.api.NetworkException;
025    import com.kallasoft.smugmug.api.util.APIUtils;
026    
027    /**
028     * Class used to represent the base implementation of a {@link Method} in the
029     * kallasoft SmugMug Java API.
030     * <p>
031     * All custom <em>method</em> implementations should extend this class and
032     * provide the additional detail required to support the new method.
033     * <p>
034     * The {@link #executeImpl(String, String[])} implementation represents the
035     * actual implementation of calling the SmugMug JSON API and getting a
036     * JSON-formatted reply that it returns to the caller.
037     * {@link #executeImpl(String, String[])} is the nuts and bolts of the
038     * implementation to {@link Method#execute(String, String[])} but it cannot
039     * return a specific {@link AbstractResponse} instance custom for each
040     * {@link Method} implementation, so instead it returns the JSON-formatted reply
041     * for the caller to parse into an appropriate {@link AbstractResponse} instance
042     * and return.
043     * <p>
044     * The implementation of {@link #executeImpl(String, String[])} is fully
045     * optimized according to the <a
046     * href="http://jakarta.apache.org/httpcomponents/httpclient-3.x/performance.html">HttpClient
047     * Performance Guide</a> to provide the best performance, memory management,
048     * error reporting and resource-cleanup. Because of this, implementors are
049     * encouraged to extend this class when implementing their own methods to take
050     * advantage of it.
051     * 
052     * @author Riyad Kalla
053     */
054    public abstract class AbstractMethod implements Method {
055            private static final Logger logger = LoggerFactory
056                            .getLogger(AbstractMethod.class);
057    
058            private String methodName;
059    
060            private String[] arguments;
061    
062            /**
063             * Construct a method that represents the given method name which takes the
064             * given arguments.
065             * 
066             * @param methodName
067             *            The name of the SmugMug JSON API method that this
068             *            <em>Method</em> represents.
069             * @param arguments
070             *            The names of the arguments that this method accepts.
071             * 
072             * @throws IllegalArgumentException
073             *             if methodName is <code>null</code> or empty.
074             */
075            public AbstractMethod(String methodName, String[] arguments)
076                            throws IllegalArgumentException {
077                    if (APIUtils.isEmpty(methodName))
078                            throw new IllegalArgumentException(
079                                            "methodName cannot be null or empty");
080    
081                    this.methodName = methodName;
082                    this.arguments = arguments;
083    
084                    logger.debug("Created Instance of Method: {}", getMethodName());
085            }
086    
087            public String getMethodName() {
088                    return methodName;
089            }
090    
091            public String[] getArguments() {
092                    return arguments;
093            }
094    
095            /**
096             * Used to take the given {@link PostMethod} and setup all the name/value
097             * pairs for it before it is executed.
098             * <p>
099             * Any <code>null</code> values for <code>arguments</code> or
100             * <code>argumentValues</code> cause that pair of name/value to be
101             * ignored. For example, consider the pairs:
102             * <ul>
103             * <li>(1, bob)</li>
104             * <li>(2, mary)</li>
105             * <li>(3, <code>null</code>)</li>
106             * <li>(<code>null</code>, scott)</li>
107             * </ul>. The first 2 name/value pairs will be included in the HTTP POST,
108             * but the 3rd and 4th will be ignored because either their name or value is
109             * <code>null</code>.
110             * <p>
111             * If more arguments exist than values, all the arguments without matching
112             * values will be ignored.
113             * 
114             * @param postMethod
115             *            The {@link PostMethod} that will be setup with all the
116             *            appropriate name/value pairs in order to call the SmugMug
117             *            server.
118             * @param argumentValues
119             *            The values to pass to the API call for each of the arguments
120             *            it takes.
121             * 
122             * @throws IllegalArgumentException
123             *             if postMethod is <code>null</code>.
124             * @throws IllegalArgumentException
125             *             if methodName is <code>null</code> or empty.
126             * @throws IllegalArgumentException
127             *             if either arguments or argumentValues is <code>null</code>.
128             * @throws IllegalArgumentException
129             *             if arguments.length &lt; argumentValues.length
130             */
131            protected void setupPostParameters(PostMethod postMethod,
132                            String[] argumentValues) throws IllegalArgumentException {
133                    String methodName = getMethodName();
134                    String[] arguments = getArguments();
135    
136                    if (postMethod == null)
137                            throw new IllegalArgumentException("postMethod cannot be null");
138    
139                    if (APIUtils.isEmpty(methodName))
140                            throw new IllegalArgumentException(
141                                            "methodName cannot be null or empty");
142    
143                    if (arguments == null || argumentValues == null)
144                            throw new IllegalArgumentException("Neither arguments ["
145                                            + arguments + "] or argumentValues [" + argumentValues
146                                            + "] can be null");
147    
148                    if (arguments.length < argumentValues.length)
149                            throw new IllegalArgumentException(
150                                            "arguments.length must be >= argumentValues.length. If arguments.length is larger than argumentValues.length, the extra arguments are ignored.");
151    
152                    logger
153                                    .debug("\t\tAdding argument name=[method] value=[{}]",
154                                                    methodName);
155    
156                    postMethod.addParameter("method", methodName);
157    
158                    for (int i = 0; i < arguments.length; i++) {
159                            String argument = arguments[i];
160                            String argumentValue = null;
161    
162                            if (i < argumentValues.length)
163                                    argumentValue = argumentValues[i];
164                            else {
165                                    /*
166                                     * We had more arguments than values, so stop building the
167                                     * string
168                                     */
169                                    break;
170                            }
171    
172                            /* If we have valid name/value pair, add it to the HTTP POST method */
173                            if (!APIUtils.isEmpty(argument) && !APIUtils.isEmpty(argumentValue)) {
174                                    logger.debug("\t\tAdding argument name=[{}] value=[{}]",
175                                                    argument, argumentValue);
176    
177                                    postMethod.addParameter(argument, argumentValue);
178                            }
179                    }
180            }
181    
182            /**
183             * The core implementation of the kallasoft SmugMug Java API that takes the
184             * given URL and argument values, and communicates with the SmugMug JSON API
185             * then returns the JSON-formatted reply to the caller.
186             * <p>
187             * Custom {@link Method} implementors are encouraged to use this method
188             * internally, parsing the JSON reply into the appropriate
189             * {@link AbstractResponse} then returning that to their caller.
190             * <p>
191             * The implementation of this method is fully optimized according to the <a
192             * href="http://jakarta.apache.org/httpcomponents/httpclient-3.x/performance.html">HttpClient
193             * Performance Guide</a> to provide the best performance, memory
194             * management, error reporting and resource-cleanup. Because of this,
195             * implementors are encouraged to extend this class when implementing their
196             * own methods to take advantage of it.
197             * 
198             * @param url
199             *            The URL of the SmugMug API to call.
200             * @param argumentValues
201             *            The values to pass to the API call for each of the arguments
202             *            it takes.
203             * 
204             * @return a JSON-formatted reply from the SmugMug JSON API.
205             * 
206             * @throws IllegalArgumentException
207             *             if url is null or empty.
208             * @throws NetworkException
209             *             if anything other than a HTTP status code of 200 is returned
210             *             from the server. This is not the same as a network exception
211             *             and normally represents a failure on the SmugMug server side.
212             * @throws NetworkException
213             *             if any network-based error occurs while trying to communicate
214             *             with the SmugMug server. This can wrap many different kinds
215             *             of more specific network-based exceptions. The handler can
216             *             call <code>getCause()</code> to retrieve a more specific
217             *             reason for the exception if necessary.
218             */
219            protected String executeImpl(String url, String[] argumentValues)
220                            throws IllegalArgumentException, NetworkException {
221                    logger.debug("Executing Method {} Using Service URL {}",
222                                    getMethodName(), url);
223    
224                    if (APIUtils.isEmpty(url))
225                            throw new IllegalArgumentException("url [" + url
226                                            + "] cannot be null or empty");
227    
228                    String response = null;
229                    PostMethod postMethod = null;
230    
231                    try {
232                            /* Setup communication to the server */
233                            postMethod = new PostMethod(url);
234    
235                            /* Setup the HTTP POST parameters */
236                            postMethod.setRequestHeader("User-Agent", APIConstants.USER_AGENT);
237                            setupPostParameters(postMethod, argumentValues);
238    
239                            logger.trace("\tExecuting HTTP POST...");
240    
241                            /* Perform the communication with the server */
242                            int code = APIConstants.HTTP_CLIENT.executeMethod(postMethod);
243    
244                            logger.debug("\tReceived HTTP status code {}", Integer
245                                            .valueOf(code));
246    
247                            /*
248                             * Make sure the server responded with a status code of 200 (OK),
249                             * otherwise we won't likely have a valid response from it to parse.
250                             */
251                            if (code == HttpStatus.SC_OK) {
252                                    response = IOUtils.toString(postMethod
253                                                    .getResponseBodyAsStream(), postMethod
254                                                    .getResponseCharSet());
255    
256                                    logger.debug("\tRead response, was {} bytes long", Integer
257                                                    .valueOf(response.length()));
258                            } else {
259                                    String message = "An HTTP status code of ["
260                                                    + code
261                                                    + "] was returned from the server, but a status code of 200 (OK) was expected. Something may be wrong with the SmugMug server at the moment or the network path to the SmugMug server.";
262                                    NetworkException ne = new NetworkException(message);
263                                    logger.error(message, ne);
264                                    throw ne;
265                            }
266                    } catch (Exception e) {
267                            RuntimeException re = new RuntimeException(e);
268                            logger
269                                            .error(
270                                                            "An error occured while trying to execute the HTTP Post call",
271                                                            re);
272                            throw re;
273                    } finally {
274                            logger.trace("\tAttempting to cleanup network resources...");
275    
276                            /*
277                             * We must release the method no matter what, otherwise the
278                             * connection is not released back to the HttpConnectionManager. Do
279                             * this as safely as possible.
280                             */
281                            if (postMethod != null) {
282                                    try {
283                                            postMethod.releaseConnection();
284                                            logger.trace("\t\tNetwork resources cleaned up!");
285                                    } catch (Exception e) {
286                                            String message = "Unable to release the connection used by the HttpClient instance. This should not happen.";
287                                            RuntimeException re = new RuntimeException(message, e);
288                                            logger.error(message, re);
289                                            throw re;
290                                    }
291                            }
292                    }
293    
294                    logger.debug("\tReturning JSON response to caller...");
295                    return response;
296            }
297    }
