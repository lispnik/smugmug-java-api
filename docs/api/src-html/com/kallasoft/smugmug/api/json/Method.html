015    package com.kallasoft.smugmug.api.json;
016    
017    /**
018     * Interface used to define a method in the kallasoft SmugMug Java API.
019     * <p>
020     * A method typically represents one of the <a
021     * href="http://smugmug.jot.com/WikiHome/API">SmugMug JSON API methods</a> but
022     * implementors are free to build on top of the <code>Method</code>
023     * implementation support to build any functionality they would like.
024     * <p>
025     * The SmugMug JSON API uses the standard HTTP protocol to communicate,
026     * therefore all values passed to the kallasoft SmugMug Java API must be
027     * converted to a <code>String</code> representation as the last step before
028     * it can be included in the HTTP POST to the SmugMug server. Because of this,
029     * the base implementation of {@link #execute(String, String[])} in all methods
030     * of the kallasoft SmugMug Java API have been normalized to accept a
031     * <code>String[]</code> of values, in order, for each value included in the
032     * request identified by the {@link #getArguments()} method.
033     * <p>
034     * While the kallasoft SmugMug Java API attempts to provide convenience methods
035     * for most all methods to make using them easier, implements are free to add
036     * more convenience methods that can take other values and ultimately handle
037     * converting them to <code>String</code>s before making the call to
038     * {@link #execute(String, String[])}.
039     * 
040     * @author Riyad Kalla
041     */
042    public interface Method {
043            /**
044             * Used to get the name of the SmugMug JSON API method that this
045             * <code>Method</code> represents and will call when executed. For
046             * example, <code>smugmug.login.withPassword</code>.
047             * 
048             * @return the name of the SmugMug JSON API method that this method
049             *         represents and will call when executed.
050             */
051            public String getMethodName();
052    
053            /**
054             * Used to get a list of the argument names for this method. For example,
055             * <code>APIKey, SessionID and AlbumID</code>.
056             * 
057             * @return a list of the argument names for this method.
058             */
059            public String[] getArguments();
060    
061            /**
062             * Used to execute a <code>Method</code> by calling the given
063             * <code>url</code> and passing the given <code>argumentValues</code>
064             * for each of the arguments this method defines (identified by
065             * {@link #getArguments()})
066             * <p>
067             * While implementations of <code>Method</code> may provide additional
068             * wrapper methods that make it easier to use the API, this method defines
069             * the central entry point into the kallasoft SmugMug Java API where the
070             * communication with the SmugMug server is initiated and result parsing
071             * begun.
072             * 
073             * @param url
074             *            The URL of the SmugMug API to call.
075             * @param argumentValues
076             *            The values to pass to the API call for each of the arguments
077             *            it takes.
078             * 
079             * @return a {@link AbstractResponse} implementation that represents the
080             *         JSON-formatted reply from the SmugMug JSON API call.
081             */
082            public AbstractResponse execute(String url, String[] argumentValues);
083    }
