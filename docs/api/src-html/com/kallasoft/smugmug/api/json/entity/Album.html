008    /**
009     * Class used to define an album and all the properties it may or may not
010     * contain.
011     * <p>
012     * This class also defines all the argument value constants for
013     * <code>TemplateID</code>, <code>SortMethod</code>,
014     * <code>SortDirection</code> and <code>DefaultColor</code> that can be
015     * passed or returned to or from an {@link Album} or {@link AlbumTemplate}.
016     * <p>
017     * <strong>NOTE</strong>: Boolean values of <code>0</code> and <code>1</code>
018     * are represented by <code>String</code>s because the SmugMug API only
019     * accepts the numerical values for <code>false</code> and <code>true</code>
020     * respectively and cannot accept textual representations (e.g. "True", "Yes",
021     * "false", "no", etc.) which is what {@link Boolean#toString()} returns when
022     * invoked. So the literal values of <code>0</code> and <code>1</code> are
023     * represented by <code>String</code>s to control that.
024     * 
025     * @author Riyad Kalla
026     */

244            /**
245             * Construct a new album by safely parsing all known album values out of the
246             * given {@link JSONObject}.
247             * <p>
248             * If a property is not available from the {@link JSONObject}, it's value
249             * is simply set to <code>null</code> and parsing continues.
250             * 
251             * @param albumObject
252             *            The {@link JSONObject} that will be queried for all the known
253             *            properties for this album.
254             * 
255             * @throws JSONException
256             *             if any JSONObject-related exception occurs.
257             */
258            public Album(JSONObject albumObject) throws JSONException {
259                    /*
260                     * If albumObject is null, leave everything set to the default value of
261                     * null and return.
262                     */
263                    if (albumObject == null)
264                            return;
265    
266                    /* Parse required fields */
267                    id = JSONUtils.getIntegerSafely(albumObject, "id");
268                    albumKey = JSONUtils.getStringSafely(albumObject, "Key");
269                    title = JSONUtils.getStringSafely(albumObject, "Title");
270    
271                    /* Parse common fields */
272                    description = JSONUtils.getStringSafely(albumObject, "Description");
273                    keywords = JSONUtils.getStringSafely(albumObject, "Keywords");
274    
275                    if (!albumObject.isNull("Category"))
276                            category = new Category(albumObject.getJSONObject("Category"));
277    
278                    if (!albumObject.isNull("SubCategory"))
279                            subCategory = new Category(albumObject.getJSONObject("SubCategory"));
280    
281                    geography = JSONUtils.getBooleanSafely(albumObject, "Geography");
282                    albumTemplateID = JSONUtils.getIntegerSafely(albumObject,
283                                    "AlbumTemplateID");
284    
285                    /* Parse look & feel fields */
286                    exif = JSONUtils.getBooleanSafely(albumObject, "EXIF");
287                    clean = JSONUtils.getBooleanSafely(albumObject, "Clean");
288                    header = JSONUtils.getBooleanSafely(albumObject, "Header");
289                    filenames = JSONUtils.getBooleanSafely(albumObject, "Filenames");
290    
291                    if (!albumObject.isNull("Template"))
292                            templateID = JSONUtils.getIntegerSafely(albumObject
293                                            .getJSONObject("Template"), "id");
294    
295                    sortMethod = JSONUtils.getStringSafely(albumObject, "SortMethod");
296                    sortDirection = JSONUtils
297                                    .getBooleanSafely(albumObject, "SortDirection");
298                    position = JSONUtils.getIntegerSafely(albumObject, "Position");
299                    imageCount = JSONUtils.getIntegerSafely(albumObject, "ImageCount");
300    
301                    if (!albumObject.isNull("Highlight"))
302                            highlight = new Image(albumObject.getJSONObject("Highlight"));
303    
304                    squareThumbs = JSONUtils.getBooleanSafely(albumObject, "SquareThumbs");
305    
306                    /* Parse security & privacy fields */
307                    password = JSONUtils.getStringSafely(albumObject, "Password");
308                    passwordHint = JSONUtils.getStringSafely(albumObject, "PasswordHint");
309                    passworded = JSONUtils.getBooleanSafely(albumObject, "Passworded");
310                    isProtected = JSONUtils.getBooleanSafely(albumObject, "Protected");
311                    isPublic = JSONUtils.getBooleanSafely(albumObject, "Public");
312                    hideOwner = JSONUtils.getBooleanSafely(albumObject, "HideOwner");
313                    external = JSONUtils.getBooleanSafely(albumObject, "External");
314                    smugSearchable = JSONUtils.getBooleanSafely(albumObject,
315                                    "SmugSearchable");
316                    worldSearchable = JSONUtils.getBooleanSafely(albumObject,
317                                    "WorldSearchable");
318                    larges = JSONUtils.getBooleanSafely(albumObject, "Larges");
319                    xLarges = JSONUtils.getBooleanSafely(albumObject, "XLarges");
320                    x2Larges = JSONUtils.getBooleanSafely(albumObject, "X2Larges");
321                    x3Larges = JSONUtils.getBooleanSafely(albumObject, "X3Larges");
322                    originals = JSONUtils.getBooleanSafely(albumObject, "Originals");
323                    watermarking = JSONUtils.getBooleanSafely(albumObject, "Watermarking");
324    
325                    if (!albumObject.isNull("Watermark"))
326                            watermarkID = JSONUtils.getIntegerSafely(albumObject
327                                            .getJSONObject("Watermark"), "id");
328    
329                    /* Parse social fields */
330                    share = JSONUtils.getBooleanSafely(albumObject, "Share");
331                    canRank = JSONUtils.getBooleanSafely(albumObject, "CanRank");
332                    comments = JSONUtils.getBooleanSafely(albumObject, "Comments");
333                    familyEdit = JSONUtils.getBooleanSafely(albumObject, "FamilyEdit");
334                    friendEdit = JSONUtils.getBooleanSafely(albumObject, "FriendEdit");
335    
336                    /* Parse community fields */
337                    if (!albumObject.isNull("Community"))
338                            communityID = JSONUtils.getIntegerSafely(albumObject
339                                            .getJSONObject("Community"), "id");
340    
341                    /* Parse printing & sales fields */
342                    printable = JSONUtils.getBooleanSafely(albumObject, "Printable");
343                    proofDays = JSONUtils.getIntegerSafely(albumObject, "ProofDays");
344                    backprinting = JSONUtils.getStringSafely(albumObject, "Backprinting");
345                    defaultColor = JSONUtils.getBooleanSafely(albumObject, "DefaultColor");
346    
347                    /* Parse sharpening fields */
348                    unsharpAmount = JSONUtils.getFloatSafely(albumObject, "UnsharpAmount");
349                    unsharpRadius = JSONUtils.getFloatSafely(albumObject, "UnsharpRadius");
350                    unsharpThreshold = JSONUtils.getFloatSafely(albumObject,
351                                    "UnsharpThreshold");
352                    unsharpSigma = JSONUtils.getFloatSafely(albumObject, "UnsharpSigma");
353    
354                    /* Parse misc fields */
355                    lastUpdated = JSONUtils.getStringSafely(albumObject, "LastUpdated");
356            }

358            /**
359             * Construct a new album with all the given property values.
360             * 
361             * @param id
362             *            The ID of the album.
363             * @param albumKey
364             *            The security key for the album.
365             * @param title
366             *            The title of the album.
367             * @param description
368             *            A description of the album.
369             * @param keywords
370             *            The keywords associated with the album.
371             * @param category
372             *            The category that this album belongs to.
373             * @param subCategory
374             *            The subcategory that this album belongs to.
375             * @param geography
376             *            <code>true</code> if the album allows the use of geo-tagging
377             *            images (location), otherwise <code>false</code>.
378             * @param albumTemplateID
379             *            The ID of the album template used to configure this album.
380             * @param exif
381             *            <code>true</code> if the album allows accessing the EXIF
382             *            image data, otherwise <code>false</code>.
383             * @param clean
384             *            <code>true</code> if the album should display using the
385             *            "clean" style, otherwise <code>false</code>.
386             * @param header
387             *            <code>true</code> if the album will display a custom header,
388             *            otherwise <code>false</code>.
389             * @param filenames
390             *            <code>true</code> if the album allows filenames to be shown
391             *            if there is no caption, otherwise <code>false</code>.
392             * @param templateID
393             *            The ID of the template that will be used to display the album.
394             *            <p>
395             *            This can be:
396             *            <ul>
397             *            <li>Viewer Choice (Recommended!): 0</li>
398             *            <li>SmugMug: 3</li>
399             *            <li>Traditional: 4</li>
400             *            <li>All Thumbs: 7</li>
401             *            <li>Slideshow: 8</li>
402             *            <li>Journal: 9</li>
403             *            <li>SmugMug Small: 10</li>
404             *            <li>Filmstrip: 11</li>
405             *            </ul>
406             * @param sortMethod
407             *            The sort method used for the images in this album.
408             *            <p>
409             *            This can be:
410             *            <ul>
411             *            <li>"Position" (Sorts by user-specified position)</li>
412             *            <li>"Caption" (Sorts by the image captions)</li>
413             *            <li>"FileName" (Sorts by the filename of each photo)</li>
414             *            <li>"Date" (Sorts by the date uploaded to SmugMug)</li>
415             *            <li>"DateTime" (Sorts by the date last modified, as told by
416             *            EXIF data - Many files don't have this field set correctly)</li>
417             *            <li>"DateTimeOriginal" (Sorts by the date taken, as told by
418             *            EXIF data - Many cameras don't report this properly)</li>
419             *            </ul>
420             * @param sortDirection
421             *            <code>false</code> (0) for ascending and <code>true</code>
422             *            (1) for descending.
423             * @param position
424             *            The position of the album within it's category (starting with
425             *            1).
426             * @param imageCount
427             *            The number of images contained in the album.
428             * @param highlight
429             *            The {@link Image} (must have ID and ImageKey set) that will be
430             *            highlighted for this album.
431             * @param squareThumbs
432             *            <code>true</code> if thumbnails (Tiny and Thumbnail sizes)
433             *            are generated in a square aspect ratio, <code>false</code>
434             *            in order to generate them using the original aspect ratio.
435             * @param password
436             *            The password used to password-protect the album if it is
437             *            password protected.
438             * @param passwordHint
439             *            The password hint for the password if there is one.
440             * @param passworded
441             *            <code>true</code> if the album is password protected ,
442             *            otherwise <code>false</code>.
443             * @param isProtected
444             *            <code>true</code> if the album is using heavy image
445             *            protection, otherwise <code>false</code>.
446             * @param isPublic
447             *            <code>true</code> if the album is publicly accessible,
448             *            otherwise <code>false</code>.
449             * @param hideOwner
450             *            <code>true</code> if the album should mask the URL and name
451             *            of the image owner, otherwise <code>false</code>.
452             * @param external
453             *            <code>true</code> if the album allows external linking,
454             *            otherwise <code>false</code>.
455             * @param smugSearchable
456             *            <code>true</code> if the album is SmugMug-searchable,
457             *            otherwise <code>false</code>.
458             * @param worldSearchable
459             *            <code>true</code> if the album is world-searchable (e.g.
460             *            Google, Yahoo, etc.), otherwise <code>false</code>.
461             * @param larges
462             *            <code>true</code> if the album allows viewing of large-size
463             *            images, otherwise <code>false</code>.
464             * @param xLarges
465             *            <code>true</code> if the album allows viewing of xlarge-size
466             *            images, otherwise <code>false</code>.
467             * @param x2Larges
468             *            <code>true</code> if the album allows viewing of
469             *            xxlarge-size images, otherwise <code>false</code>.
470             * @param x3Larges
471             *            <code>true</code> if the album allows viewing of
472             *            xxxlarge-size images, otherwise <code>false</code>.
473             * @param originals
474             *            <code>true</code> if the album allows the image's originals
475             *            to be seen, otherwise <code>false</code>.
476             * @param watermarking
477             *            <code>true</code> if the album uses watermarking, otherwise
478             *            <code>false</code>.
479             * @param watermarkID
480             *            The ID of the watermark that will be used on the images.
481             * @param share
482             *            <code>true</code> if the album enables the "Easy Sharing"
483             *            button and features, otherwise <code>false</code>.
484             * @param canRank
485             *            <code>true</code> if the album can be ranked, otherwise
486             *            <code>false</code>.
487             * @param comments
488             *            <code>true</code> if the album allows comments, otherwise
489             *            <code>false</code>.
490             * @param familyEdit
491             *            <code>true</code> if the album allows captions and keywords
492             *            to be edited by "family", otherwise <code>false</code>.
493             * @param friendEdit
494             *            <code>true</code> if the album allows captions and keywords
495             *            to be edited by "friends", otherwise <code>false</code>.
496             * @param communityID
497             *            The ID of the community that this album belongs to.
498             * @param printable
499             *            <code>true</code> if the album allows prints to be
500             *            purchased, otherwise <code>false</code>.
501             * @param proofDays
502             *            The number of days to delay printing for this album.
503             * @param backprinting
504             *            The optional text that can be printed on the backs of the
505             *            prints.
506             * @param defaultColor
507             *            <code>true</code> if the album defaults to using "True" or
508             *            "AutoColor" when it's in the cart, otherwise
509             *            <code>false</code>.
510             * @param unsharpAmount
511             *            How weak or strong the sharpness mask will be. A higher value
512             *            will result in a more stark sharpness (Default 0.200). This
513             *            only applies to resized versions of the original image (Large,
514             *            XLarge, etc.) See <a
515             *            href="http://smugmug.com/help/display-quality">http://smugmug.com/help/display-quality</a>
516             *            for more information.
517             * @param unsharpRadius
518             *            The radius on the sharpening effect. Larger values lead to
519             *            more defined sharpening (Default 1.000). This only applies to
520             *            resized versions of the original image (Large, XLarge, etc.)
521             *            See <a
522             *            href="http://smugmug.com/help/display-quality">http://smugmug.com/help/display-quality</a>
523             *            for more information.
524             * @param unsharpThreshold
525             *            The number of adjacent pixels that are considered an "edge"
526             *            before performing a sharpen (Default 0.050). This only applies
527             *            to resized versions of the original image (Large, XLarge,
528             *            etc.) See <a
529             *            href="http://smugmug.com/help/display-quality">http://smugmug.com/help/display-quality</a>
530             *            for more information.
531             * @param unsharpSigma
532             *            (Default 1.000)
533             * @param lastUpdated
534             *            The date when the album was last updated.
535             */
