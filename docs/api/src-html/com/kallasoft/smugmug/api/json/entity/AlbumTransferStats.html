051            /**
052             * Construct a new album transfer stat object by safely parsing all known
053             * album values (and any contained image transfer stat objects if available)
054             * out of the given {@link JSONObject}.
055             * <p>
056             * If a property is not available from the {@link JSONObject}, it's value
057             * is simply set to <code>null</code> and ignored and parsing continues.
058             * 
059             * @param albumObject
060             *            The {@link JSONObject} that will be queried for all the album
061             *            transfer stats.
062             * 
063             * @throws IllegalArgumentException
064             *             if <code>albumObject</code> is null.
065             * 
066             * @throws JSONException
067             *             if any JSONObject-related exception occurs.
068             */
069            public AlbumTransferStats(JSONObject albumObject)
070                            throws IllegalArgumentException, JSONException {
071                    if (albumObject == null)
072                            throw new IllegalArgumentException("albumObject cannot be null");
073    
074                    /* Parse all the values */
075                    id = JSONUtils.getIntegerSafely(albumObject, "id");
076                    bytes = JSONUtils.getIntegerSafely(albumObject, "Bytes");
077                    tiny = JSONUtils.getIntegerSafely(albumObject, "Tiny");
078                    thumb = JSONUtils.getIntegerSafely(albumObject, "Thumb");
079                    small = JSONUtils.getIntegerSafely(albumObject, "Small");
080                    medium = JSONUtils.getIntegerSafely(albumObject, "Medium");
081                    large = JSONUtils.getIntegerSafely(albumObject, "Large");
082                    xLarge = JSONUtils.getIntegerSafely(albumObject, "XLarge");
083                    x2Large = JSONUtils.getIntegerSafely(albumObject, "X2Large");
084                    x3Large = JSONUtils.getIntegerSafely(albumObject, "X3Large");
085                    original = JSONUtils.getFloatSafely(albumObject, "Original");
086                    video320 = JSONUtils.getFloatSafely(albumObject, "Video320");
087                    video640 = JSONUtils.getFloatSafely(albumObject, "Video640");
088                    video960 = JSONUtils.getFloatSafely(albumObject, "Video960");
089                    video1280 = JSONUtils.getFloatSafely(albumObject, "Video1280");
090    
091                    /* Now process the images in this album if there are any */
092                    if (!albumObject.isNull("Images")) {
093                            JSONArray imageArray = albumObject.getJSONArray("Images");
094    
095                            for (int j = 0, jLength = imageArray.length(); j < jLength; j++)
096                                    imageTransferStatsList.add(new ImageTransferStats(imageArray
097                                                    .getJSONObject(j)));
098                    }
099            }
100    
101            /**
102             * Construct an object with the given album transfer stats.
103             * 
104             * @param id
105             *            The ID of the album.
106             * @param bytes
107             *            The number of bytes transferred for this album.
108             * @param tiny
109             *            The number of times tiny-sized images have been viewed for
110             *            this album.
111             * @param thumb
112             *            The number of times thumb-sized images have been viewed for
113             *            this album.
114             * @param small
115             *            The number of times small-sized images have been viewed for
116             *            this album.
117             * @param medium
118             *            The number of times medium-sized images have been viewed for
119             *            this album.
120             * @param large
121             *            The number of times large-sized images have been viewed for
122             *            this album.
123             * @param xLarge
124             *            The number of times xlarge-sized images have been viewed for
125             *            this album.
126             * @param x2Large
127             *            The number of times xxlarge-sized images have been viewed for
128             *            this album.
129             * @param x3Large
130             *            The number of times xxxlarge-sized images have been viewed for
131             *            this album.
132             * @param original
133             *            The number of times original-sized images have been viewed for
134             *            this album.
135             * @param video320
136             *            The number of times 320-pixel sized movies have been viewed
137             *            for this album.
138             * @param video640
139             *            The number of times 640-pixel sized movies have been viewed
140             *            for this album.
141             * @param video960
142             *            The number of times 960-pixel sized movies have been viewed
143             *            for this album.
144             * @param video1280
145             *            The number of times 1280-pixel sized movies have been viewed
146             *            for this album.
147             * @param imageTransferStatsList
148             *            A list of image transfer stat objects that represent the
149             *            images in this album, if available.
150             */
151            public AlbumTransferStats(Integer id, Integer bytes, Integer tiny,
152                            Integer thumb, Integer small, Integer medium, Integer large,
153                            Integer xLarge, Integer x2Large, Integer x3Large, Float original,
154                            Float video320, Float video640, Float video960, Float video1280,
155                            List<ImageTransferStats> imageTransferStatsList) {
156                    this.id = id;
157                    this.bytes = bytes;
158                    this.tiny = tiny;
159                    this.thumb = thumb;
160                    this.small = small;
161                    this.medium = medium;
162                    this.large = large;
163                    this.original = original;
164                    this.video320 = video320;
165                    this.video640 = video640;
166                    this.video960 = video960;
167                    this.video1280 = video1280;
168                    this.imageTransferStatsList = imageTransferStatsList;
169            }

171            @Override
172            public String toString() {
173                    return Album.class.getName() + "[id=" + getID() + ", bytes="
174                                    + getBytes() + ", tiny=" + getTiny() + ", thumb=" + getThumb()
175                                    + ", small=" + getSmall() + ", medium=" + getMedium()
176                                    + ", large=" + getLarge() + ", xLarge=" + getXLarge()
177                                    + ", x2Large=" + getX2Large() + ", x3Large=" + getX3Large()
178                                    + ", original=" + getOriginal() + ", video320=" + getVideo320()
179                                    + ", video640=" + getVideo640() + ", video960=" + getVideo960()
180                                    + ", video1280=" + getVideo1280()
181                                    + ", imageTransferStatsListSize="
182                                    + getImageTransferStatsList().size() + "]";
183            }
