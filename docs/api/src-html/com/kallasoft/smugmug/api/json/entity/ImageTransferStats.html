001    package com.kallasoft.smugmug.api.json.entity;
002    
003    import org.json.JSONObject;
004    
005    import com.kallasoft.smugmug.api.json.util.JSONUtils;
006    
007    /**
008     * Class used to define an image and all the transfer statistics that can be
009     * reported for it.
010     * 
011     * @author Riyad Kalla
012     */
013    public class ImageTransferStats {
014            private Integer id;
015    
016            private Integer bytes;
017    
018            private Integer tiny;
019    
020            private Integer thumb;
021    
022            private Integer small;
023    
024            private Integer medium;
025    
026            private Integer large;
027    
028            private Integer xLarge;
029    
030            private Integer x2Large;
031    
032            private Integer x3Large;
033    
034            private Float original;
035    
036            private Float video320;
037    
038            private Float video640;
039    
040            private Float video960;
041    
042            private Float video1280;
043    
044            /**
045             * Construct a new image transfer stat object by safely parsing all known
046             * image values out of the given {@link JSONObject}.
047             * <p>
048             * If a property is not available from the {@link JSONObject}, it's value
049             * is simply set to <code>null</code> and ignored and parsing continues.
050             * 
051             * @param imageObject
052             *            The {@link JSONObject} that will be queried for all the image
053             *            transfer stats.
054             * 
055             * @throws IllegalArgumentException
056             *             if <code>imageObject</code> is null.
057             */
058            public ImageTransferStats(JSONObject imageObject)
059                            throws IllegalArgumentException {
060                    if (imageObject == null)
061                            throw new IllegalArgumentException("imageObject cannot be null");
062    
063                    id = JSONUtils.getIntegerSafely(imageObject, "id");
064                    bytes = JSONUtils.getIntegerSafely(imageObject, "Bytes");
065                    tiny = JSONUtils.getIntegerSafely(imageObject, "Tiny");
066                    thumb = JSONUtils.getIntegerSafely(imageObject, "Thumb");
067                    small = JSONUtils.getIntegerSafely(imageObject, "Small");
068                    medium = JSONUtils.getIntegerSafely(imageObject, "Medium");
069                    large = JSONUtils.getIntegerSafely(imageObject, "Large");
070                    xLarge = JSONUtils.getIntegerSafely(imageObject, "XLarge");
071                    x2Large = JSONUtils.getIntegerSafely(imageObject, "X2Large");
072                    x3Large = JSONUtils.getIntegerSafely(imageObject, "X3Large");
073                    original = JSONUtils.getFloatSafely(imageObject, "Original");
074                    video320 = JSONUtils.getFloatSafely(imageObject, "Video320");
075                    video640 = JSONUtils.getFloatSafely(imageObject, "Video640");
076                    video960 = JSONUtils.getFloatSafely(imageObject, "Video960");
077                    video1280 = JSONUtils.getFloatSafely(imageObject, "Video1280");
078            }
079    
080            /**
081             * Construct an object with the given image transfer stats.
082             * 
083             * @param id
084             *            The ID of the image.
085             * @param bytes
086             *            The number of bytes transferred for this image.
087             * @param tiny
088             *            The number of times tiny-sized images have been viewed for
089             *            this image.
090             * @param thumb
091             *            The number of times thumb-sized images have been viewed for
092             *            this image.
093             * @param small
094             *            The number of times small-sized images have been viewed for
095             *            this image.
096             * @param medium
097             *            The number of times medium-sized images have been viewed for
098             *            this image.
099             * @param large
100             *            The number of times large-sized images have been viewed for
101             *            this image.
102             * @param xLarge
103             *            The number of times xlarge-sized images have been viewed for
104             *            this image.
105             * @param x2Large
106             *            The number of times xxlarge-sized images have been viewed for
107             *            this image.
108             * @param x3Large
109             *            The number of times xxxlarge-sized images have been viewed for
110             *            this image.
111             * @param original
112             *            The number of times original-sized images have been viewed for
113             *            this image.
114             * @param video320
115             *            The number of times 320-pixel sized movies have been viewed
116             *            for this album.
117             * @param video640
118             *            The number of times 640-pixel sized movies have been viewed
119             *            for this album.
120             * @param video960
121             *            The number of times 960-pixel sized movies have been viewed
122             *            for this album.
123             * @param video1280
124             *            The number of times 1280-pixel sized movies have been viewed
125             *            for this album.
126             */
127            public ImageTransferStats(Integer id, Integer bytes, Integer tiny,
128                            Integer thumb, Integer small, Integer medium, Integer large,
129                            Integer xLarge, Integer x2Large, Integer x3Large, Float original,
130                            Float video320, Float video640, Float video960, Float video1280) {
131                    this.id = id;
132                    this.bytes = bytes;
133                    this.tiny = tiny;
134                    this.thumb = thumb;
135                    this.small = small;
136                    this.medium = medium;
137                    this.large = large;
138                    this.original = original;
139                    this.video320 = video320;
140                    this.video640 = video640;
141                    this.video960 = video960;
142                    this.video1280 = video1280;
143            }
144    
145            @Override
146            public String toString() {
147                    return ImageTransferStats.class.getName() + "[id=" + getID()
148                                    + ", bytes=" + getBytes() + ", tiny=" + getTiny() + ", thumb="
149                                    + getThumb() + ", small=" + getSmall() + ", medium="
150                                    + getMedium() + ", large=" + getLarge() + ", xLarge="
151                                    + getXLarge() + ", x2Large=" + getX2Large() + ", x3Large="
152                                    + getX3Large() + ", original=" + getOriginal() + ", video320="
153                                    + getVideo320() + ", video640=" + getVideo640() + ", video960="
154                                    + getVideo960() + ", video1280=" + getVideo1280() + "]";
155            }
156
