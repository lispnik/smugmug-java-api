022    /**
023     * This method will change the settings on an album specified by AlbumID.
024     * <p>
025     * The session must be established with a valid login. All settings are
026     * optional, if omitted, the current value for that setting is preserved.
027     * 
028     * @author Riyad Kalla
029     * @version 1.2.0
030     * @see <a
031     *      href="http://smugmug.jot.com/WikiHome/1.2.0/smugmug.albums.changeSettings">smugmug.albums.changeSettings
032     *      API Doc</a>
033     */
034    public class ChangeSettings extends AbstractMethod {
035            /* Parameters all copied from the Album class, which is the master copy */
036    
037            /**
038             * Defines the SmugMug JSON API method name that will be called.
039             */
040            public static final String METHOD_NAME = "smugmug.albums.changeSettings";
041

048             * "SortMethod", "SortDirection", "Position", "HighlightID", "SquareThumbs",

056            public static final String[] ARGUMENTS = { "APIKey", "SessionID",
057                            "AlbumID", "Title", "Description", "Keywords", "CategoryID",
058                            "SubCategoryID", "Geography", "AlbumTemplateID", "EXIF", "Clean",
059                            "Header", "Filenames", "TemplateID", "SortMethod", "SortDirection",
060                            "Position", "HighlightID", "SquareThumbs", "Password",
061                            "PasswordHint", "Protected", "Public", "HideOwner", "External",
062                            "SmugSearchable", "WorldSearchable", "Larges", "XLarges",
063                            "X2Larges", "X3Larges", "Originals", "Watermarking", "WatermarkID",
064                            "Share", "CanRank", "Comments", "FamilyEdit", "FriendEdit",

070             * Construct a new method instance that can be executed.
071             */
072            public ChangeSettings() {
073                    this(METHOD_NAME, ARGUMENTS);
074            }
075    
076            /**
077             * Construct a new method instance that can be executed with the given
078             * arguments.
079             * 
080             * @param methodName
081             *            The name of the SmugMug JSON API method that this
082             *            <em>Method</em> represents.
083             * @param arguments
084             *            The names of the arguments that this method accepts.
085             */
086            public ChangeSettings(String methodName, String[] arguments) {
087                    super(methodName, arguments);
088            }
089    
090            /**
091             * Used to execute the smugmug.albums.changeSettings method, changing any
092             * settings specified and leaving any settings not specified as they are.
093             * 
094             * @param url
095             *            The URL of the SmugMug server to communicate with.
096             * @param argumentValues
097             *            The argument values to pass to this method.
098             * 
099             * @return the response returned after the settings have changed.
100             */
101            public ChangeSettingsResponse execute(String url, String[] argumentValues) {
102                    return new ChangeSettingsResponse(executeImpl(url, argumentValues));
103            }

105            /**
106             * Convenience method used to execute the smugmug.albums.changeSettings
107             * method.
108             * <p>
109             * This method performs necessary conversions on all the argument values
110             * before calling {@link #execute(String, String[])}.
111             * 
112             * @param url
113             *            The URL of the SmugMug server to communicate with.
114             * @param apiKey
115             *            The API Key to use. API keys are issued by SmugMug.
116             * @param sessionID
117             *            The logged in SessionID that represents the user's session.
118             * @param albumID
119             *            The ID of the album to change the settings for.
120             * @param title
121             *            The title of the album.
122             * @param description
123             *            A description of the album.
124             * @param keywords
125             *            The keywords associated with the album.
126             * @param categoryID
127             *            The ID of the category that this album belongs to.
128             * @param subCategoryID
129             *            The ID of the subcategory that this album belongs to.
130             * @param geography
131             *            <code>true</code> if the album allows the use of geo-tagging
132             *            images (location), otherwise <code>false</code>.
133             * @param albumTemplateID
134             *            The ID of the album template used to configure this album.
135             * @param exif
136             *            <code>true</code> if the album allows accessing the EXIF
137             *            image data, otherwise <code>false</code>.
138             * @param clean
139             *            <code>true</code> if the album should display using the
140             *            "clean" style, otherwise <code>false</code>.
141             * @param header
142             *            <code>true</code> if the album will display a custom header,
143             *            otherwise <code>false</code>.
144             * @param filenames
145             *            <code>true</code> if the album allows filenames to be shown
146             *            if there is no caption, otherwise <code>false</code>.
147             * @param templateID
148             *            The ID of the template that will be used to display the album.
149             *            <p>
150             *            This can be:
151             *            <ul>
152             *            <li>Viewer Choice (Recommended!): 0</li>
153             *            <li>SmugMug: 3</li>
154             *            <li>Traditional: 4</li>
155             *            <li>All Thumbs: 7</li>
156             *            <li>Slideshow: 8</li>
157             *            <li>Journal: 9</li>
158             *            <li>SmugMug Small: 10</li>
159             *            <li>Filmstrip: 11</li>
160             *            </ul>
161             * @param sortMethod
162             *            The sort method used for the images in this album.
163             *            <p>
164             *            This can be:
165             *            <ul>
166             *            <li>"Position" (Sorts by user-specified position)</li>
167             *            <li>"Caption" (Sorts by the image captions)</li>
168             *            <li>"FileName" (Sorts by the filename of each photo)</li>
169             *            <li>"Date" (Sorts by the date uploaded to SmugMug)</li>
170             *            <li>"DateTime" (Sorts by the date last modified, as told by
171             *            EXIF data - Many files don't have this field set correctly)</li>
172             *            <li>"DateTimeOriginal" (Sorts by the date taken, as told by
173             *            EXIF data - Many cameras don't report this properly)</li>
174             *            </ul>
175             * @param sortDirection
176             *            <code>false</code> (0) for ascending and <code>true</code>
177             *            (1) for descending.
178             * @param position
179             *            The position of the album within it's category (starting with
180             *            1).
181             * @param highlightID
182             *            The ID of the image that will be highlighted (displayed) for
183             *            this album.
184             * @param squareThumbs
185             *            <code>true</code> if thumbnails (Tiny and Thumbnail sizes)
186             *            are generated in a square aspect ratio, <code>false</code>
187             *            in order to generate them using the original aspect ratio.
188             * @param password
189             *            The password used to password-protect the album if it is
190             *            password protected.
191             * @param passwordHint
192             *            The password hint for the password if there is one.
193             * @param isProtected
194             *            <code>true</code> if the album is using heavy image
195             *            protection, otherwise <code>false</code>.
196             * @param isPublic
197             *            <code>true</code> if the album is publicly accessible,
198             *            otherwise <code>false</code>.
199             * @param hideOwner
200             *            <code>true</code> if the album should mask the URL and name
201             *            of the image owner, otherwise <code>false</code>.
202             * @param external
203             *            <code>true</code> if the album allows external linking,
204             *            otherwise <code>false</code>.
205             * @param smugSearchable
206             *            <code>true</code> if the album is SmugMug-searchable,
207             *            otherwise <code>false</code>.
208             * @param worldSearchable
209             *            <code>true</code> if the album is world-searchable (e.g.
210             *            Google, Yahoo, etc.), otherwise <code>false</code>.
211             * @param larges
212             *            <code>true</code> if the album allows viewing of large-size
213             *            images, otherwise <code>false</code>.
214             * @param xLarges
215             *            <code>true</code> if the album allows viewing of xlarge-size
216             *            images, otherwise <code>false</code>.
217             * @param x2Larges
218             *            <code>true</code> if the album allows viewing of
219             *            xxlarge-size images, otherwise <code>false</code>.
220             * @param x3Larges
221             *            <code>true</code> if the album allows viewing of
222             *            xxxlarge-size images, otherwise <code>false</code>.
223             * @param originals
224             *            <code>true</code> if the album allows the image's originals
225             *            to be seen, otherwise <code>false</code>.
226             * @param watermarking
227             *            <code>true</code> if the album uses watermarking, otherwise
228             *            <code>false</code>.
229             * @param watermarkID
230             *            The ID of the watermark that will be used on the images.
231             * @param share
232             *            <code>true</code> if the album enables the "Easy Sharing"
233             *            button and features, otherwise <code>false</code>.
234             * @param canRank
235             *            <code>true</code> if the album can be ranked, otherwise
236             *            <code>false</code>.
237             * @param comments
238             *            <code>true</code> if the album allows comments, otherwise
239             *            <code>false</code>.
240             * @param familyEdit
241             *            <code>true</code> if the album allows captions and keywords
242             *            to be edited by "family", otherwise <code>false</code>.
243             * @param friendEdit
244             *            <code>true</code> if the album allows captions and keywords
245             *            to be edited by "friends", otherwise <code>false</code>.
246             * @param communityID
247             *            The ID of the community that this album belongs to.
248             * @param printable
249             *            <code>true</code> if the album allows prints to be
250             *            purchased, otherwise <code>false</code>.
251             * @param proofDays
252             *            The number of days to delay printing for this album.
253             * @param backprinting
254             *            The optional text that can be printed on the backs of the
255             *            prints.
256             * @param defaultColor
257             *            <code>true</code> if the album defaults to using "True" or
258             *            "AutoColor" when it's in the cart, otherwise
259             *            <code>false</code>.
260             * @param unsharpAmount
261             *            How weak or strong the sharpness mask will be. A higher value
262             *            will result in a more stark sharpness (Default 0.200). This
263             *            only applies to resized versions of the original image (Large,
264             *            XLarge, etc.) See <a
265             *            href="http://smugmug.com/help/display-quality">http://smugmug.com/help/display-quality</a>
266             *            for more information.
267             * @param unsharpRadius
268             *            The radius on the sharpening effect. Larger values lead to
269             *            more defined sharpening (Default 1.000). This only applies to
270             *            resized versions of the original image (Large, XLarge, etc.)
271             *            See <a
272             *            href="http://smugmug.com/help/display-quality">http://smugmug.com/help/display-quality</a>
273             *            for more information.
274             * @param unsharpThreshold
275             *            The number of adjacent pixels that are considered an "edge"
276             *            before performing a sharpen (Default 0.050). This only applies
277             *            to resized versions of the original image (Large, XLarge,
278             *            etc.) See <a
279             *            href="http://smugmug.com/help/display-quality">http://smugmug.com/help/display-quality</a>
280             *            for more information.
281             * @param unsharpSigma
282             *            (Default 1.000)
283             * 
284             * @return the response returned after the settings have changed.
285             * 
286             * @see #execute(String, String[])
287             */
288            public ChangeSettingsResponse execute(String url, String apiKey,
289                            String sessionID, Integer albumID, String title,
290                            String description, String keywords, Integer categoryID,
291                            Integer subCategoryID, Boolean geography, Integer albumTemplateID,
292                            Boolean exif, Boolean clean, Boolean header, Boolean filenames,
293                            Integer templateID, String sortMethod, Boolean sortDirection,
294                            Integer position, Integer highlightID, Boolean squareThumbs,

296                            Boolean isPublic, Boolean hideOwner, Boolean external,
297                            Boolean smugSearchable, Boolean worldSearchable, Boolean larges,
298                            Boolean xLarges, Boolean x2Larges, Boolean x3Larges,
299                            Boolean originals, Boolean watermarking, Integer watermarkID,
300                            Boolean share, Boolean canRank, Boolean comments,
301                            Boolean familyEdit, Boolean friendEdit, Integer communityID,
302                            Boolean printable, Integer proofDays, String backprinting,
303                            Boolean defaultColor, Float unsharpAmount, Float unsharpRadius,
304                            Float unsharpThreshold, Float unsharpSigma) {

310                                    APIUtils.toString(clean), APIUtils.toString(header),
311                                    APIUtils.toString(filenames), APIUtils.toString(templateID),
312                                    sortMethod, APIUtils.toString(sortDirection),
313                                    APIUtils.toString(position), APIUtils.toString(highlightID),
314                                    APIUtils.toString(squareThumbs), password, passwordHint,
315                                    APIUtils.toString(isProtected), APIUtils.toString(isPublic),
316                                    APIUtils.toString(hideOwner), APIUtils.toString(external),
317                                    APIUtils.toString(smugSearchable),
318                                    APIUtils.toString(worldSearchable), APIUtils.toString(larges),
319                                    APIUtils.toString(xLarges), APIUtils.toString(x2Larges),
320                                    APIUtils.toString(x3Larges), APIUtils.toString(originals),
321                                    APIUtils.toString(watermarking),
322                                    APIUtils.toString(watermarkID), APIUtils.toString(share),
323                                    APIUtils.toString(canRank), APIUtils.toString(comments),
324                                    APIUtils.toString(familyEdit), APIUtils.toString(friendEdit),
325                                    APIUtils.toString(communityID), APIUtils.toString(printable),
326                                    APIUtils.toString(proofDays), APIUtils.toString(backprinting),
327                                    APIUtils.toString(defaultColor),
328                                    APIUtils.toString(unsharpAmount),
329                                    APIUtils.toString(unsharpRadius),
330                                    APIUtils.toString(unsharpThreshold),
331                                    APIUtils.toString(unsharpSigma) });
332            }

336             * smugmug.albums.changeSettings method call.

340             * was successful or not.
341             * 
342             * @author Riyad Kalla
343             * @version 1.2.0
344             */

352                     *            SmugMug API call.
353                     * 
354                     * @throws RuntimeJSONException
355                     *             if an error occurs while parsing the JSON response text.
356                     */
357                    public ChangeSettingsResponse(String responseText)
358                                    throws RuntimeJSONException {
359                            /* Parse base response and any error if necessary */
360                            super(responseText);
361                    }

366                                            + isError() + "]";
367                    }
368            }
