366            /**
367             * Class used to represent the response for the smugmug.albums.create method
368             * call.
369             * 
370             * @author Riyad Kalla
371             * @version 1.2.0
372             */
373            public class CreateResponse extends AbstractResponse {
374                    private Integer albumID;
375    
376                    private String albumKey;
377    
378                    /**
379                     * Construct a response by parsing the necessary values out of the JSON
380                     * response text.
381                     * 
382                     * @param responseText
383                     *            The JSON-formatted response text that came back from the
384                     *            SmugMug API call.
385                     * 
386                     * @throws RuntimeJSONException
387                     *             if an error occurs while parsing the JSON response text.
388                     */
389                    public CreateResponse(String responseText) throws RuntimeJSONException {
390                            /* Parse base response and any error if necessary */
391                            super(responseText);
392    
393                            /* If there was an error, or the message is empty, do nothing */
394                            if (isError() || APIUtils.isEmpty(responseText))
395                                    return;
396    
397                            JSONObject responseObject = null;
398    
399                            try {
400                                    responseObject = new JSONObject(responseText);
401                                    JSONObject albumObject = responseObject.getJSONObject("Album");
402    
403                                    albumID = JSONUtils.getIntegerSafely(albumObject, "id");
404                                    albumKey = JSONUtils.getStringSafely(albumObject, "Key");
405                            } catch (JSONException e) {
406                                    RuntimeJSONException rje = new RuntimeJSONException(e);
407                                    logger.error("An error occured parsing the JSON response", rje);
408                                    throw rje;
409                            }
410                    }
411    
412                    @Override
413                    public String toString() {
414                            return CreateResponse.class.getName() + "[isError=" + isError()
415                                            + ", albumID=" + getAlbumID() + ", albumKey="
416                                            + getAlbumKey() + "]";
417                    }
418    
419                    /**
420                     * Used to get the ID of the album that was created.
421                     * 
422                     * @return the ID of the album that was created.
423                     */
424                    public Integer getAlbumID() {
425                            return albumID;
426                    }
427    
428                    /**
429                     * Used to get the security key for the album that was created.
430                     * 
431                     * @return the security key for the album that was created.
432                     */
433                    public String getAlbumKey() {
434                            return albumKey;
435                    }
436            }
