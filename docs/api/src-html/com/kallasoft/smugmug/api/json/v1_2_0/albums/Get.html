032    /**
033     * This method returns a list of albums for a given user (either via SessionID
034     * or (optional) NickName). If the session is logged in as the requested user, a
035     * complete list of albums is returned. Otherwise, only the public albums are
036     * returned.
037     * 
038     * @author Riyad Kalla
039     * @version 1.2.0
040     * @see <a
041     *      href="http://smugmug.jot.com/WikiHome/1.2.0/smugmug.albums.get">smugmug.albums.get
042     *      API Doc</a>
043     */
044    public class Get extends AbstractMethod {
045            /**
046             * Defines the SmugMug JSON API method name that will be called.
047             */
048            public static final String METHOD_NAME = "smugmug.albums.get";
049    
050            /**
051             * Defines all the arguments this method takes.
052             * <p>
053             * Values are: "APIKey", "SessionID", "NickName", "Heavy", "SitePassword"
054             */
055            public static final String[] ARGUMENTS = { "APIKey", "SessionID",
056                            "NickName", "Heavy", "SitePassword" };
057    
058            private static final Logger logger = LoggerFactory.getLogger(Get.class);
059    
060            /**
061             * Construct a new method instance that can be executed.
062             */
063            public Get() {
064                    this(METHOD_NAME, ARGUMENTS);
065            }
066    
067            /**
068             * Construct a new method instance that can be executed with the given
069             * arguments.
070             * 
071             * @param methodName
072             *            The name of the SmugMug JSON API method that this
073             *            <em>Method</em> represents.
074             * @param arguments
075             *            The names of the arguments that this method accepts.
076             */
077            public Get(String methodName, String[] arguments) {
078                    super(methodName, arguments);
079            }
080    
081            /**
082             * Used to execute the smugmug.albums.get method, returning a list of albums
083             * for the user identified by the Session ID or the given Nick Name.
084             * 
085             * @param url
086             *            The URL of the SmugMug server to communicate with.
087             * @param argumentValues
088             *            The argument values to pass to this method.
089             * 
090             * @return the response that includes a list of the albums for the user
091             *         identified by the given Session ID or Nick Name.
092             */
093            public GetResponse execute(String url, String[] argumentValues) {
094                    return new GetResponse(executeImpl(url, argumentValues));
095            }
096    
097            /**
098             * Convenience method used to execute the smugmug.albums.get method.
099             * <p>
100             * This method performs necessary conversions on all the argument values
101             * before calling {@link #execute(String, String[])}.
102             * 
103             * @param url
104             *            The URL of the SmugMug server to communicate with.
105             * @param apiKey
106             *            The API Key to use. API keys are issued by SmugMug.
107             * @param sessionID
108             *            The logged in SessionID that represents the user's session.
109             * @param heavy
110             *            A boolean argument (0 or 1) indicating if you would like
111             *            <em>heavy</em> results returned. Heavy results include much
112             *            more detail about each album and take longer to process when
113             *            requested.
114             * 
115             * @return the response that includes a list of the albums for the user
116             *         identified by the given Session ID.
117             * 
118             * @see #execute(String, String[])
119             */
120            public GetResponse execute(String url, String apiKey, String sessionID,
121                            Boolean heavy) {
122                    return execute(url, new String[] { apiKey, sessionID, null,
123                                    APIUtils.toString(heavy) });
124            }
125    
126            /**
127             * Convenience method used to execute the smugmug.albums.get method.
128             * <p>
129             * This method performs necessary conversions on all the argument values
130             * before calling {@link #execute(String, String[])}.
131             * 
132             * @param url
133             *            The URL of the SmugMug server to communicate with.
134             * @param apiKey
135             *            The API Key to use. API keys are issued by SmugMug.
136             * @param sessionID
137             *            The logged in SessionID that represents the user's session.
138             * @param nickName
139             *            A nick name for a user that you wish to load the album list
140             *            for. If the album list for the given nick name is not public,
141             *            a sitePassword must be provided.
142             * @param heavy
143             *            A boolean argument (0 or 1) indicating if you would like
144             *            <em>heavy</em> results returned. Heavy results include much
145             *            more detail about each album and take longer to process when
146             *            requested.
147             * @param sitePassword
148             *            The password for a given password-protected account identified
149             *            by <code>nickName</code>.
150             * 
151             * @return the response that includes a list of the albums for the user
152             *         identified by the given Session ID or Nick Name.
153             * 
154             * @see #execute(String, String[])
155             */
156            public GetResponse execute(String url, String apiKey, String sessionID,
157                            String nickName, Boolean heavy, String sitePassword) {
158                    return execute(url, new String[] { apiKey, sessionID, nickName,
159                                    APIUtils.toString(heavy), sitePassword });
160            }
161    
162            /**
163             * Class used to represent the response for the smugmug.albums.get method
164             * call.
165             * 
166             * @author Riyad Kalla
167             * @version 1.2.0
168             */
169            public class GetResponse extends AbstractResponse {
170                    private List<Album> albumList = new ArrayList<Album>();
171    
172                    /**
173                     * Construct a response by parsing the necessary values out of the JSON
174                     * response text.
175                     * 
176                     * @param responseText
177                     *            The JSON-formatted response text that came back from the
178                     *            SmugMug API call.
179                     * 
180                     * @throws RuntimeJSONException
181                     *             if an error occurs while parsing the JSON response text.
182                     */
183                    public GetResponse(String responseText) throws RuntimeJSONException {
184                            /* Parse base response and any error if necessary */
185                            super(responseText);
186    
187                            /* If there was an error, or the message is empty, do nothing */
188                            if (isError() || APIUtils.isEmpty(responseText))
189                                    return;
190    
191                            JSONObject responseObject = null;
192    
193                            try {
194                                    responseObject = new JSONObject(responseText);
195                                    JSONArray albumArray = responseObject.getJSONArray("Albums");
196    
197                                    for (int i = 0, length = albumArray.length(); i < length; i++)
198                                            albumList.add(new Album(albumArray.getJSONObject(i)));
199                            } catch (JSONException e) {
200                                    RuntimeJSONException rje = new RuntimeJSONException(e);
201                                    logger.error("An error occured parsing the JSON response", rje);
202                                    throw rje;
203                            }
204                    }
205    
206                    @Override
207                    public String toString() {
208                            return GetResponse.class.getName() + "[isError=" + isError()
209                                            + ", albumListSize=" + albumList.size() + "]";
210                    }
211    
212                    /**
213                     * Used to get a list of the albums that were returned.
214                     * 
215                     * @return a list of the albums that were returned.
216                     */
217                    public List<Album> getAlbumList() {
218                            return albumList;
219                    }
220            }
221    }
