022    import com.kallasoft.smugmug.api.json.AbstractMethod;

028    /**
029     * Gets transfer statistics for the given Album during the given Month and Year.
030     * SmugMug often only keeps the last few months (current + 2 previous).
031     * <p>
032     * A float is provided for Originals because videos often are only partially
033     * viewed.
034     * <p>
035     * If 'Heavy' is set to '1', an array of images with each image transfer stat
036     * (just like smugmug.images.getStats) is provided for each album in addition to
037     * the album transfer stats.
038     * 
039     * @author Riyad Kalla
040     * @version 1.2.0
041     * @see <a
042     *      href="http://smugmug.jot.com/WikiHome/1.2.0/smugmug.albums.getStats">smugmug.albums.getStats
043     *      API Doc</a>
044     */
045    public class GetStats extends AbstractMethod {
046            /**
047             * Defines the SmugMug JSON API method name that will be called.
048             */
049            public static final String METHOD_NAME = "smugmug.albums.getStats";
050    
051            /**
052             * Defines all the arguments this method takes.
053             * <p>
054             * Values are: "APIKey", "SessionID", "AlbumID", "Month", "Year", "Heavy"
055             */
056            public static final String[] ARGUMENTS = { "APIKey", "SessionID",
057                            "AlbumID", "Month", "Year", "Heavy" };
058    
059            private static final Logger logger = LoggerFactory
060                            .getLogger(GetStats.class);
061    
062            /**
063             * Construct a new method instance that can be executed.
064             */
065            public GetStats() {
066                    this(METHOD_NAME, ARGUMENTS);
067            }
068    
069            /**
070             * Construct a new method instance that can be executed with the given
071             * arguments.
072             * 
073             * @param methodName
074             *            The name of the SmugMug JSON API method that this
075             *            <em>Method</em> represents.
076             * @param arguments
077             *            The names of the arguments that this method accepts.
078             */
079            public GetStats(String methodName, String[] arguments) {
080                    super(methodName, arguments);
081            }
082    
083            /**
084             * Used to execute the smugmug.albums.getStats method, returning the
085             * transfer stats for an album, and if the response is heavy, returning
086             * transfer stats for all the images in the album as well.
087             * 
088             * @param url
089             *            The URL of the SmugMug server to communicate with.
090             * @param argumentValues
091             *            The argument values to pass to this method.
092             * 
093             * @return the response that includes the transfer stats for the given album
094             *         and if the response is heavy, a list of transfer stats for each
095             *         image in the album.
096             */
097            public GetStatsResponse execute(String url, String[] argumentValues) {
098                    return new GetStatsResponse(executeImpl(url, argumentValues));
099            }
100    
101            /**
102             * Convenience method used to execute the smugmug.albums.getStats method.
103             * <p>
104             * This method performs necessary conversions on all the argument values
105             * before calling {@link #execute(String, String[])}.
106             * 
107             * @param url
108             *            The URL of the SmugMug server to communicate with.
109             * @param apiKey
110             *            The API Key to use. API keys are issued by SmugMug.
111             * @param sessionID
112             *            The logged in SessionID that represents the user's session.
113             * @param albumID
114             *            The ID of the album to get the transfer stats for.
115             * @param month
116             *            The number of the month (1-12) to get the transfer stats for
117             *            the given album.
118             * @param year
119             *            The year (e.g. 2007) to get the transfer stats for the given
120             *            album.
121             * 
122             * @return the response that includes the transfer stats for the given
123             *         album.
124             * 
125             * @see #execute(String, String[])
126             */
127            public GetStatsResponse execute(String url, String apiKey,
128                            String sessionID, Integer albumID, Integer month, Integer year) {
129                    return execute(url, new String[] { apiKey, sessionID,
130                                    APIUtils.toString(albumID), APIUtils.toString(month),
131                                    APIUtils.toString(year) });
132            }
133    
134            /**
135             * Convenience method used to execute the smugmug.albums.getStats method.
136             * <p>
137             * This method performs necessary conversions on all the argument values
138             * before calling {@link #execute(String, String[])}.
139             * 
140             * @param url
141             *            The URL of the SmugMug server to communicate with.
142             * @param apiKey
143             *            The API Key to use. API keys are issued by SmugMug.
144             * @param sessionID
145             *            The logged in SessionID that represents the user's session.
146             * @param albumID
147             *            The ID of the album to get the transfer stats for.
148             * @param month
149             *            The number of the month (1-12) to get the transfer stats for
150             *            the given album.
151             * @param year
152             *            The year (e.g. 2007) to get the transfer stats for the given
153             *            album.
154             * @param heavy
155             *            Set to <code>true</code> to return heavy results, including
156             *            transfer stats for all the images in the album as well,
157             *            otherwise set to <code>false</code>.
158             * 
159             * @return the response that includes the transfer stats for the given album
160             *         and if the response is heavy, a list of transfer stats for each
161             *         image in the album.
162             * 
163             * @see #execute(String, String[])
164             */
165            public GetStatsResponse execute(String url, String apiKey,
166                            String sessionID, Integer albumID, Integer month, Integer year,
167                            Boolean heavy) {
168                    return execute(url, new String[] { apiKey, sessionID,
169                                    APIUtils.toString(albumID), APIUtils.toString(month),
170                                    APIUtils.toString(year), APIUtils.toString(heavy) });
171            }
172    
173            /**
174             * Class used to represent the response for the smugmug.albums.getStats
175             * method call.
176             * 
177             * @author Riyad Kalla
178             * @version 1.2.0
179             */
180            public class GetStatsResponse extends AbstractResponse {
181                    private AlbumTransferStats albumTransferStats;
182    
183                    /**
184                     * Construct a response by parsing the necessary values out of the JSON
185                     * response text.
186                     * 
187                     * @param responseText
188                     *            The JSON-formatted response text that came back from the
189                     *            SmugMug API call.
190                     * 
191                     * @throws RuntimeJSONException
192                     *             if an error occurs while parsing the JSON response text.
193                     */
194                    public GetStatsResponse(String responseText)
195                                    throws RuntimeJSONException {
196                            /* Parse base response and any error if necessary */
197                            super(responseText);
198    
199                            /* If there was an error, or the message is empty, do nothing */
200                            if (isError() || APIUtils.isEmpty(responseText))
201                                    return;
202    
203                            JSONObject responseObject = null;
204    
205                            try {
206                                    responseObject = new JSONObject(responseText);
207    
208                                    albumTransferStats = new AlbumTransferStats(responseObject
209                                                    .getJSONObject("Album"));
210                            } catch (JSONException e) {
211                                    RuntimeJSONException rje = new RuntimeJSONException(e);
212                                    logger.error("An error occured parsing the JSON response", rje);
213                                    throw rje;
214                            }
215                    }
216    
217                    @Override
218                    public String toString() {
219                            return GetStatsResponse.class.getName() + "[isError=" + isError()
220                                            + ", albumTransferStats=" + getAlbumTransferStats() + "]";
221                    }
222    
223                    /**
224                     * Used to get the transfer statistics for this album.
225                     * 
226                     * @return the transfer statistics for this album.
227                     */
228                    public AlbumTransferStats getAlbumTransferStats() {
229                            return albumTransferStats;
230                    }
231            }
