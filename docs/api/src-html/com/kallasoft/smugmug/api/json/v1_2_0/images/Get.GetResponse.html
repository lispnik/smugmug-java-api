032    /**
033     * This method will fetch all the ImageIDs for the album specified by AlbumID.
034     * <p>
035     * The album must be owned by the session holder, or else be public (if
036     * password-protected, a password must be provided) to return results, otherwise
037     * an "invalid user" faultCode will result.
038     * 
039     * @author Riyad Kalla
040     * @version 1.2.0
041     * @see <a
042     *      href="http://smugmug.jot.com/WikiHome/1.2.0/smugmug.images.get">smugmug.images.get
043     *      API Doc</a>
044     */
045    public class Get extends AbstractMethod {
046            /**
047             * Defines the SmugMug JSON API method name that will be called.
048             */
049            public static final String METHOD_NAME = "smugmug.images.get";
050    
051            /**
052             * Defines all the arguments this method takes.
053             * <p>
054             * Values are: "APIKey", "SessionID", "AlbumID", "AlbumKey", "Heavy",
055             * "Password", "SitePassword"
056             */
057            public static final String[] ARGUMENTS = { "APIKey", "SessionID",
058                            "AlbumID", "AlbumKey", "Heavy", "Password", "SitePassword" };
059    
060            private static final Logger logger = LoggerFactory.getLogger(Get.class);
061    
062            /**
063             * Construct a new method instance that can be executed.
064             */
065            public Get() {
066                    this(METHOD_NAME, ARGUMENTS);
067            }
068    
069            /**
070             * Construct a new method instance that can be executed with the given
071             * arguments.
072             * 
073             * @param methodName
074             *            The name of the SmugMug JSON API method that this
075             *            <em>Method</em> represents.
076             * @param arguments
077             *            The names of the arguments that this method accepts.
078             */
079            public Get(String methodName, String[] arguments) {
080                    super(methodName, arguments);
081            }
082    
083            /**
084             * Used to execute the smugmug.images.get method, returning a list of image
085             * IDs for the given album.
086             * 
087             * @param url
088             *            The URL of the SmugMug server to communicate with.
089             * @param argumentValues
090             *            The argument values to pass to this method.
091             * 
092             * @return the response that includes a list of the image IDs for the given
093             *         album.
094             */
095            public GetResponse execute(String url, String[] argumentValues) {
096                    return new GetResponse(executeImpl(url, argumentValues));
097            }
098    
099            /**
100             * Convenience method used to execute the smugmug.images.get method.
101             * <p>
102             * This method performs necessary conversions on all the argument values
103             * before calling {@link #execute(String, String[])}.
104             * 
105             * @param url
106             *            The URL of the SmugMug server to communicate with.
107             * @param apiKey
108             *            The API Key to use. API keys are issued by SmugMug.
109             * @param sessionID
110             *            The logged in SessionID that represents the user's session.
111             * @param albumID
112             *            The ID of the album whose image IDs will be returned.
113             * @param albumKey
114             *            The security key for the album.
115             * 
116             * @return the response that includes a list of the image IDs for the given
117             *         album.
118             * 
119             * @see #execute(String, String[])
120             */
121            public GetResponse execute(String url, String apiKey, String sessionID,
122                            Integer albumID, String albumKey) {
123                    return execute(url, new String[] { apiKey, sessionID,
124                                    APIUtils.toString(albumID), albumKey });
125            }

128             * Convenience method used to execute the smugmug.images.get method.

155                            Integer albumID, String albumKey, Boolean heavy) {

158                                                    albumKey, APIUtils.toString(heavy) });
159            }

161            /**
162             * Convenience method used to execute the smugmug.images.get method.

166             * 
167             * @param url

190             * 
191             * @see #execute(String, String[])

193            public GetResponse execute(String url, String apiKey, String sessionID,
194                            Integer albumID, String albumKey, Boolean heavy, String password,
195                            String sitePassword) {
196                    return execute(url, new String[] { apiKey, sessionID,
197                                    APIUtils.toString(albumID), albumKey, APIUtils.toString(heavy),
198                                    password, sitePassword });
199            }

201            /**
202             * Class used to represent the response for the smugmug.images.get method
203             * call.
204             * 
205             * @author Riyad Kalla
206             * @version 1.2.0
207             */
208            public class GetResponse extends AbstractResponse {
209                    private List<Image> imageList = new ArrayList<Image>();
210    
211                    /**
212                     * Construct a response by parsing the necessary values out of the JSON
213                     * response text.
214                     * 
215                     * @param responseText
216                     *            The JSON-formatted response text that came back from the
217                     *            SmugMug API call.
218                     * 
219                     * @throws RuntimeJSONException
220                     *             if an error occurs while parsing the JSON response text.
221                     */
222                    public GetResponse(String responseText) throws RuntimeJSONException {
223                            /* Parse base response and any error if necessary */
224                            super(responseText);
225    
226                            /* If there was an error, or the message is empty, do nothing */
227                            if (isError() || APIUtils.isEmpty(responseText))
228                                    return;
229    
230                            JSONObject responseObject = null;
231    
232                            try {
233                                    responseObject = new JSONObject(responseText);
234    
235                                    JSONArray imageArray = responseObject.getJSONArray("Images");
236    
237                                    for (int i = 0, length = imageArray.length(); i < length; i++)
238                                            imageList.add(new Image(imageArray.getJSONObject(i)));
239                            } catch (JSONException e) {
240                                    RuntimeJSONException rje = new RuntimeJSONException(e);
241                                    logger.error("An error occured parsing the JSON response", rje);
242                                    throw rje;
243                            }
244                    }
245    
246                    @Override
247                    public String toString() {
248                            return GetResponse.class.getName() + "[isError=" + isError()
249                                            + ", imageListSize=" + getImageList().size() + "]";
250                    }
251    
252                    /**
253                     * Used to get a list of the images that were returned.
254                     * 
255                     * @return a list of the images that were returned.
256                     */
257                    public List<Image> getImageList() {
258                            return imageList;
259                    }
260            }
