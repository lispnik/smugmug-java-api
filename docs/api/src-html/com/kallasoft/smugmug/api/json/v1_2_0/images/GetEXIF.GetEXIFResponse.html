028    /**
029     * This method will return camera and photograph details about the image
030     * specified by ImageID.
031     * <p>
032     * The album must be owned by the session holder, or else be public (if
033     * password-protected, a password must be provided) to return results, otherwise
034     * an "invalid user" faultCode will result.
035     * <p>
036     * Additionally, the album owner must have specified that EXIF data is allowed.
037     * <p>
038     * Note that many photos have no EXIF data, so an empty or partially returned
039     * result is very normal.
040     * 
041     * @author Riyad Kalla
042     * @version 1.2.0
043     * @see <a
044     *      href="http://smugmug.jot.com/WikiHome/1.2.0/smugmug.images.getEXIF">smugmug.images.getEXIF
045     *      API Doc</a>
046     */
047    public class GetEXIF extends AbstractMethod {
048            /**
049             * Defines the SmugMug JSON API method name that will be called.
050             */
051            public static final String METHOD_NAME = "smugmug.images.getEXIF";
052    
053            /**
054             * Defines all the arguments this method takes.
055             * <p>
056             * Values are: "APIKey", "SessionID", "ImageID", "ImageKey", "Password",
057             * "SitePassword"
058             */
059            public static final String[] ARGUMENTS = { "APIKey", "SessionID",
060                            "ImageID", "ImageKey", "Password", "SitePassword" };
061    
062            private static final Logger logger = LoggerFactory.getLogger(GetEXIF.class);
063    
064            /**
065             * Construct a new method instance that can be executed.
066             */
067            public GetEXIF() {
068                    this(METHOD_NAME, ARGUMENTS);
069            }
070    
071            /**
072             * Construct a new method instance that can be executed with the given
073             * arguments.
074             * 
075             * @param methodName
076             *            The name of the SmugMug JSON API method that this
077             *            <em>Method</em> represents.
078             * @param arguments
079             *            The names of the arguments that this method accepts.
080             */
081            public GetEXIF(String methodName, String[] arguments) {
082                    super(methodName, arguments);
083            }
084    
085            /**
086             * Used to execute the smugmug.images.getEXIF method, returning all the EXIF
087             * data from the image.
088             * 
089             * @param url
090             *            The URL of the SmugMug server to communicate with.
091             * @param argumentValues
092             *            The argument values to pass to this method.
093             * 
094             * @return the response that includes all the EXIF data from the image.
095             */
096            public GetEXIFResponse execute(String url, String[] argumentValues) {
097                    return new GetEXIFResponse(executeImpl(url, argumentValues));
098            }
099    
100            /**
101             * Convenience method used to execute the smugmug.images.getEXIF method.
102             * <p>
103             * This method performs necessary conversions on all the argument values
104             * before calling {@link #execute(String, String[])}.
105             * 
106             * @param url
107             *            The URL of the SmugMug server to communicate with.
108             * @param apiKey
109             *            The API Key to use. API keys are issued by SmugMug.
110             * @param sessionID
111             *            The logged in SessionID that represents the user's session.
112             * @param imageID
113             *            The ID of the image whose EXIF data will be returned.
114             * @param imageKey
115             *            The security key for the image.
116             * 
117             * @return the response that includes all the EXIF data from the image.
118             * 
119             * @see #execute(String, String[])
120             */
121            public GetEXIFResponse execute(String url, String apiKey, String sessionID,
122                            Integer imageID, String imageKey) {
123                    return execute(url, new String[] { apiKey, sessionID,
124                                    APIUtils.toString(imageID), imageKey });
125            }

225                    /**
226                     * Construct a response by parsing the necessary values out of the JSON
227                     * response text.
228                     * 
229                     * @param responseText
230                     *            The JSON-formatted response text that came back from the
231                     *            SmugMug API call.
232                     * 
233                     * @throws RuntimeJSONException
234                     *             if an error occurs while parsing the JSON response text.
235                     */
236                    public GetEXIFResponse(String responseText) throws RuntimeJSONException {
237                            /* Parse base response and any error if necessary */
238                            super(responseText);
239    
240                            /* If there was an error, or the message is empty, do nothing */
241                            if (isError() || APIUtils.isEmpty(responseText))
242                                    return;
243    
244                            JSONObject responseObject = null;
245    
246                            try {
247                                    responseObject = new JSONObject(responseText);
248                                    JSONObject imageObject = responseObject.getJSONObject("Image");
249    
250                                    id = JSONUtils.getIntegerSafely(imageObject, "id");
251                                    dateTime = JSONUtils.getStringSafely(imageObject, "DateTime");
252                                    dateTimeOriginal = JSONUtils.getStringSafely(imageObject,
253                                                    "DateTimeOriginal");
254                                    dateTimeDigitized = JSONUtils.getStringSafely(imageObject,
255                                                    "DateTimeDigitized");
256                                    make = JSONUtils.getStringSafely(imageObject, "Make");
257                                    model = JSONUtils.getStringSafely(imageObject, "Model");
258                                    exposureTime = JSONUtils.getStringSafely(imageObject,
259                                                    "ExposureTime");
260                                    aperture = JSONUtils.getStringSafely(imageObject, "Aperture");
261                                    iso = JSONUtils.getIntegerSafely(imageObject, "ISO");
262                                    focalLength = JSONUtils.getStringSafely(imageObject,
263                                                    "FocalLength");
264                                    focalLengthIn35mmFilm = JSONUtils.getIntegerSafely(imageObject,
265                                                    "FocalLengthIn35mmFilm");
266                                    ccdWidth = JSONUtils.getStringSafely(imageObject, "CCDWidth");
267                                    compressedBitsPerPixel = JSONUtils.getStringSafely(imageObject,
268                                                    "CompressedBitsPerPixel");
269                                    flash = JSONUtils.getIntegerSafely(imageObject, "Flash");
270                                    metering = JSONUtils.getIntegerSafely(imageObject, "Metering");
271                                    exposureProgram = JSONUtils.getIntegerSafely(imageObject,
272                                                    "ExposureProgram");
273                                    exposureBiasValue = JSONUtils.getStringSafely(imageObject,
274                                                    "ExposureBiasValue");
275                                    exposureMode = JSONUtils.getIntegerSafely(imageObject,
276                                                    "ExposureMode");
277                                    lightSource = JSONUtils.getIntegerSafely(imageObject,
278                                                    "LightSource");
279                                    whiteBalance = JSONUtils.getIntegerSafely(imageObject,
280                                                    "WhiteBalance");
281                                    digitalZoomRatio = JSONUtils.getStringSafely(imageObject,
282                                                    "DigitalZoomRatio");
283                                    contrast = JSONUtils.getIntegerSafely(imageObject, "Contrast");
284                                    saturation = JSONUtils.getIntegerSafely(imageObject,
285                                                    "Saturation");
286                                    sharpness = JSONUtils
287                                                    .getIntegerSafely(imageObject, "Sharpness");
288                                    subjectDistance = JSONUtils.getStringSafely(imageObject,
289                                                    "SubjectDistance");
290                                    subjectDistanceRange = JSONUtils.getIntegerSafely(imageObject,
291                                                    "SubjectDistanceRange");
292                                    sensingMethod = JSONUtils.getIntegerSafely(imageObject,
293                                                    "SensingMethod");
294                                    colorSpace = JSONUtils.getStringSafely(imageObject,
295                                                    "ColorSpace");
296                                    brightness = JSONUtils.getStringSafely(imageObject,
297                                                    "Brightness");
298                            } catch (JSONException e) {
299                                    RuntimeJSONException rje = new RuntimeJSONException(e);
300                                    logger.error("An error occured parsing the JSON response", rje);
301                                    throw rje;
302                            }
303                    }
