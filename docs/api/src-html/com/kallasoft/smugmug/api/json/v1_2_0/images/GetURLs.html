028    /**
029     * This method will return all the URLs for the various sizes of the image
030     * specified by ImageID.
031     * <p>
032     * The album must be owned by the session holder, or else be public (if
033     * password-protected, a password must be provided) to return results, otherwise
034     * an "invalid user" faultCode will result.
035     * <p>
036     * Additionally, obvious restrictions on Originals and (extra)Larges apply if so
037     * set by the owner. They will return as empty strings for those URLs if they're
038     * unavailable.
039     * 
040     * @author Riyad Kalla
041     * @version 1.2.0
042     * @see <a
043     *      href="http://smugmug.jot.com/WikiHome/1.2.0/smugmug.images.getURLs">smugmug.images.getURLs
044     *      API Doc</a>
045     */

059                            "ImageID", "ImageKey", "TemplateID", "Password", "SitePassword" };

233                    /**
234                     * Construct a response by parsing the necessary values out of the JSON
235                     * response text.
236                     * 
237                     * @param responseText
238                     *            The JSON-formatted response text that came back from the
239                     *            SmugMug API call.
240                     * 
241                     * @throws RuntimeJSONException
242                     *             if an error occurs while parsing the JSON response text.
243                     */
244                    public GetURLsResponse(String responseText) throws RuntimeJSONException {
245                            /* Parse base response and any error if necessary */
246                            super(responseText);
247    
248                            /* If there was an error, or the message is empty, do nothing */
249                            if (isError() || APIUtils.isEmpty(responseText))
250                                    return;
251    
252                            JSONObject responseObject = null;
253    
254                            try {
255                                    responseObject = new JSONObject(responseText);
256                                    JSONObject imageObject = responseObject.getJSONObject("Image");
257    
258                                    id = JSONUtils.getIntegerSafely(imageObject, "id");
259                                    albumURL = JSONUtils.getStringSafely(imageObject, "AlbumURL");
260                                    tinyURL = JSONUtils.getStringSafely(imageObject, "TinyURL");
261                                    thumbURL = JSONUtils.getStringSafely(imageObject, "ThumbURL");
262                                    smallURL = JSONUtils.getStringSafely(imageObject, "SmallURL");
263                                    mediumURL = JSONUtils.getStringSafely(imageObject, "MediumURL");
264                                    largeURL = JSONUtils.getStringSafely(imageObject, "LargeURL");
265                                    xLargeURL = JSONUtils.getStringSafely(imageObject, "XLargeURL");
266                                    x2LargeURL = JSONUtils.getStringSafely(imageObject,
267                                                    "X2LargeURL");
268                                    x3LargeURL = JSONUtils.getStringSafely(imageObject,
269                                                    "X3LargeURL");
270                                    originalURL = JSONUtils.getStringSafely(imageObject,
271                                                    "OriginalURL");
272                                    video320URL = JSONUtils.getStringSafely(imageObject,
273                                                    "Video320URL");
274                                    video640URL = JSONUtils.getStringSafely(imageObject,
275                                                    "Video640URL");
276                                    video960URL = JSONUtils.getStringSafely(imageObject,
277                                                    "Video960URL");
278                                    video1280URL = JSONUtils.getStringSafely(imageObject,
279                                                    "Video1280URL");
280                            } catch (JSONException e) {
281                                    RuntimeJSONException rje = new RuntimeJSONException(e);
282                                    logger.error("An error occured parsing the JSON response", rje);
283                                    throw rje;
284                            }
285                    }

298                                            + getVideo640URL() + ", video960URL=" + getVideo960URL()
299                                            + ", video1280URL=" + getVideo1280URL() + "]";
300                    }
