031    /**
032     * This method will upload a file to the specified Album (identified by AlbumID)
033     * with the required parameters.
034     * <p>
035     * It is recommended that you use the binary upload method {@link UploadHTTPPut}
036     * instead as it's a more optimized process and doesn't require that the image
037     * be Base64 encoded before uploading it, like this method does.
038     * <p>
039     * The URL provided to methods in this class should be to the Text-based SmugMug
040     * Upload server ({@link APIVersionConstants#TEXT_UPLOAD_SERVER_URL}) or the
041     * image will not be processed.
042     * <p>
043     * The inter-relationships between the methods in this class are as follows:
044     * <ul>
045     * <li>{@link #execute(String, String[])} passes it's values directly through,
046     * as-is, to the super executeImpl method, wrapping the result in an appropriate
047     * response.</li>
048     * <li>{@link #execute(String, String, String, Integer, String, InputStream)}
049     * simply passes it's values, as-is, straight through to
050     * {@link #execute(String, String, String, Integer, String, InputStream, String, String, Double, Double, Double)},
051     * including <code>null</code> for values that are unknown.</li>
052     * <li>{@link #execute(String, String, String, Integer, String, InputStream, String, String, Double, Double, Double)}
053     * utilizes
054     * {@link #prepareUploadArgumentValues(String, String, Integer, String, InputStream, String, String, Double, Double, Double)}
055     * to take all the arguments and pre-process them and then assign them,
056     * in-order, into a <code>String[]</code> that is immediately passed off to
057     * {@link #execute(String, String[])} for execution.</li>
058     * </ul>
059     * 
060     * @author Riyad Kalla
061     * @version 1.2.0
062     * @see <a
063     *      href="http://smugmug.jot.com/WikiHome/1.2.0/smugmug.images.upload">smugmug.images.upload
064     *      API Doc</a>
065     */
066    public class Upload extends AbstractMethod {
067            /**
068             * Defines the SmugMug JSON API method name that will be called.
069             */
070            public static final String METHOD_NAME = "smugmug.images.upload";
071    
072            /**
073             * Defines all the arguments this method takes.
074             * <p>
075             * Values are: "APIKey", "SessionID", "AlbumID", "FileName", "Data",
076             * "ByteCount", "MD5Sum", "Caption", "Keywords", "Latitude", "Longitude",
077             * "Altitude"
078             */
079            public static final String[] ARGUMENTS = { "APIKey", "SessionID",
080                            "AlbumID", "FileName", "Data", "ByteCount", "MD5Sum", "Caption",
081                            "Keywords", "Latitude", "Longitude", "Altitude" };
082    
083            private static final Logger logger = LoggerFactory.getLogger(Upload.class);
084    
085            /**
086             * Construct a new method instance that can be executed.
087             */
088            public Upload() {
089                    this(METHOD_NAME, ARGUMENTS);
090            }

092            /**
093             * Construct a new method instance that can be executed with the given
094             * arguments.
095             * 
096             * @param methodName
097             *            The name of the SmugMug JSON API method that this
098             *            <em>Method</em> represents.
099             * @param arguments
100             *            The names of the arguments that this method accepts.
101             */
102            public Upload(String methodName, String[] arguments) {
103                    super(methodName, arguments);
104            }

106            /**
107             * Used to execute the smugmug.images.upload method, returning the ID of the
108             * uploaded image.
109             * <p>
110             * <strong>NOTE</strong>: This method calls directly through to
111             * AbstractMethod#executeImpl, it does not call into any other method to do
112             * pre-processing of it's values.
113             * <p>
114             * For methods that offer pre-processing conveniences, please look at the
115             * other <code>execute(...)</code> methods defined in this class and check
116             * the associated Javadoc with each method to see what it does.
117             * 
118             * @param url
119             *            The URL of the SmugMug server to communicate with.
120             * @param argumentValues
121             *            The argument values to pass to this method.
122             * 
123             * @return the response that includes the ID of the uploaded image.
124             * 
125             * @see #execute(String, String, String, Integer, String, InputStream)
126             * @see #execute(String, String, String, Integer, String, InputStream,
127             *      String, String, Double, Double, Double)
128             */
129            public UploadResponse execute(String url, String[] argumentValues) {
130                    if (!APIVersionConstants.TEXT_UPLOAD_SERVER_URL.equals(url))
131                            logger
132                                            .warn(
133                                                            "url [{}] should normally be equal to the Text-based SmugMug Upload Server URL (defined by APIVersionConstants.TEXT_UPLOAD_SERVER_URL), otherwise the uploaded image won't be processed.",
134                                                            url);
135    
136                    /*
137                     * This method has to assume that the String[] being passed in has been
138                     * setup exactly as the caller wants it to be, so no additional
139                     * pre-processing (e.g. calling one of the methods that calculates MD5,
140                     * encoding, etc.) should be performed on the contents of this array.
141                     * The arguments should be passed through immediately, as-is, to
142                     * super.executeImpl.
143                     */
144                    return new UploadResponse(executeImpl(url, argumentValues));
145            }

149             * <p>
150             * This method delegates to
151             * {@link #execute(String, String, String, Integer, String, InputStream, String, String, Double, Double, Double)}.

153             * @param url
154             *            The URL of the SmugMug server to communicate with. This
155             *            usually must be equal to the the SmugMug Upload URL, otherwise
156             *            the image will not be processed.
157             * @param apiKey

159             * @param sessionID
160             *            The logged in SessionID that represents the user's session.
161             * @param albumID
162             *            The ID of the album to upload the image into.
163             * @param fileName
164             *            The name of the file whose stream will be read for data and
165             *            uploaded.
166             * @param inputStream
167             *            The {@link InputStream} to load and upload to SmugMug.

169             * @return the response that includes the ID of the uploaded image.

171             * @see #execute(String, String[])
172             * @see #execute(String, String, String, Integer, String, InputStream,
173             *      String, String, Double, Double, Double)
174             */
175            public UploadResponse execute(String url, String apiKey, String sessionID,
176                            Integer albumID, String fileName, InputStream inputStream) {
177                    return execute(url, apiKey, sessionID, albumID, fileName, inputStream,
178                                    null, null, null, null, null);
179            }

181            /**
182             * Convenience method used to execute the smugmug.images.upload method.
183             * <p>
184             * This method will perform the following tasks automatically:
185             * <ol>
186             * <li>Load the bytes of the image from the stream, keeping track of the
187             * image's size</li>
188             * <li>Calculate the MD5 Sum of the image</li>
189             * <li>Base64-encode the image data using a highly optimized Base64
190             * encoding algorithm</li>
191             * <li>Then delegate the call to {@link #execute(String, String[])},
192             * passing all the argument values included in the method call along with
193             * Data, ByteCount and MD5Sum.</li>
194             * </ol>
195             * Because of steps 1 and 3, this method will require memory slightly more
196             * than twice the size of the image being loaded. Once to load the image
197             * data, then again to load the Base64-encoded version of it.
198             * <p>
199             * This method performs necessary conversions on all the argument values
200             * before calling {@link #execute(String, String[])}.
201             * <p>
202             * <strong>NOTE</strong>: This method uses
203             * {@link #prepareUploadArgumentValues(String, String, Integer, String, InputStream, String, String, Double, Double, Double)}
204             * to do the actual preparation of the image data before delegating to
205             * {@link #execute(String, String[])}.
206             * 
207             * @param url
208             *            The URL of the SmugMug server to communicate with. This
209             *            usually must be equal to the the SmugMug Upload URL, otherwise
210             *            the image will not be processed.
211             * @param apiKey
212             *            The API Key to use. API keys are issued by SmugMug.
213             * @param sessionID
214             *            The logged in SessionID that represents the user's session.
215             * @param albumID
216             *            The ID of the album to upload the image into.
217             * @param fileName
218             *            The name of the file whose stream will be read for data and
219             *            uploaded.
220             * @param inputStream
221             *            The {@link InputStream} to load and upload to SmugMug.
222             * @param caption
223             *            A caption for the image.
224             * @param keywords
225             *            The keywords assigned to the image.
226             * @param latitude
227             *            The latitude used to geocode the image.
228             * @param longitude
229             *            The longitude used to geocode the image.
230             * @param altitude
231             *            The altitude (in meters) used to geocode the image.
232             * 
233             * @return the response that includes the ID of the uploaded image.
234             * 
235             * @see #execute(String, String[])
236             * @see #prepareUploadArgumentValues(String, String, Integer, String,
237             *      InputStream, String, String, Double, Double, Double)
238             */
239            public UploadResponse execute(String url, String apiKey, String sessionID,
240                            Integer albumID, String fileName, InputStream inputStream,
241                            String caption, String keywords, Double latitude, Double longitude,
242                            Double altitude) {
243                    return execute(url, prepareUploadArgumentValues(apiKey, sessionID,
244                                    albumID, fileName, inputStream, caption, keywords, latitude,
245                                    longitude, altitude));
246            }

248            /**
249             * Convenience method used by this class and can be used by any sub-class to
250             * prepare the given arguments by pre-processing the image data and then
251             * placing all the resulting values into a <code>String[]</code> that can
252             * be then passed directly to {@link #execute(String, String[])} to perform
253             * the upload.
254             * <p>
255             * This method will perform the following operations automatically:
256             * <ol>
257             * <li>Read the image data from the given stream using
258             * <code>APIUtils.readStream</code></li>
259             * <li>Base64-encode the read-in image data using
260             * <code>APIUtils.base64Encode</code></li>
261             * <li>Calculate the MD5 Sum of the image data that was read in using
262             * <code>APIUtils.calculateMD5Sum</code></li>
263             * </ol>
264             * 
265             * Placing all the resulting values, and the rest of the passed in values,
266             * in-order, into a <code>String[]</code> and returning it.
267             * 
268             * @param apiKey
269             *            The API Key to use. API keys are issued by SmugMug.
270             * @param sessionID
271             *            The logged in SessionID that represents the user's session.
272             * @param albumID
273             *            The ID of the album to upload the image into.
274             * @param fileName
275             *            The name of the file whose stream will be read for data and
276             *            uploaded.
277             * @param inputStream
278             *            The {@link InputStream} to load and upload to SmugMug.
279             * @param caption
280             *            A caption for the image.
281             * @param keywords
282             *            The keywords assigned to the image.
283             * @param latitude
284             *            The latitude used to geocode the image.
285             * @param longitude
286             *            The longitude used to geocode the image.
287             * @param altitude
288             *            The altitude (in meters) used to geocode the image.
289             * 
290             * @return a <code>String[]</code> that represents all the fully prepared
291             *         argument values that can now be passed directly to the
292             *         {@link #execute(String, String[])} method.
293             * 
294             * @throws RuntimeException
295             *             if any of the image operations (readStream, base64Encode or
296             *             calculateMD5Sum) fail as defined in {@link APIUtils}. The
297             *             thrown exception will wrap the real exception that describes
298             *             the failure.
299             */
300            protected String[] prepareUploadArgumentValues(String apiKey,
301                            String sessionID, Integer albumID, String fileName,
302                            InputStream inputStream, String caption, String keywords,
303                            Double latitude, Double longitude, Double altitude)
304                            throws RuntimeException {
305                    if (inputStream == null)
306                            throw new IllegalArgumentException("inputStream cannot be null");
307    
308                    /*
309                     * Perform all the image data processing steps. Any of these can throw
310                     * an exception that will kick back out to the caller to catch.
311                     */
312                    byte[] imageData = APIUtils.readStream(inputStream);
313                    String encodedData = APIUtils.base64Encode(imageData);
314                    String md5Sum = APIUtils.calculateMD5Sum(imageData);
315    
316                    String[] argumentValues = new String[12];
317    
318                    /* Now fill out the argument values array */
319                    argumentValues[0] = apiKey;
320                    argumentValues[1] = sessionID;
321                    argumentValues[2] = APIUtils.toString(albumID);
322                    argumentValues[3] = fileName;
323                    argumentValues[4] = encodedData;
324                    argumentValues[5] = Integer.toString(imageData.length);
325                    argumentValues[6] = md5Sum;
326                    argumentValues[7] = caption;
327                    argumentValues[8] = keywords;
328                    argumentValues[9] = APIUtils.toString(latitude);
329                    argumentValues[10] = APIUtils.toString(longitude);
330                    argumentValues[11] = APIUtils.toString(altitude);
331    
332                    return argumentValues;
333            }

335            /**
336             * Class used to represent the response for the smugmug.images.upload method
337             * call.
338             * 
339             * @author Riyad Kalla
340             * @version 1.2.0
341             */
342            public class UploadResponse extends AbstractResponse {
343                    private Integer imageID;
344    
345                    private String imageKey;
346    
347                    /**
348                     * Construct a response by parsing the necessary values out of the JSON
349                     * response text.
350                     * 
351                     * @param responseText
352                     *            The JSON-formatted response text that came back from the
353                     *            SmugMug API call.
354                     * 
355                     * @throws RuntimeJSONException
356                     *             if an error occurs while parsing the JSON response text.
357                     */
358                    public UploadResponse(String responseText) throws RuntimeJSONException {
359                            /* Parse base response and any error if necessary */
360                            super(responseText);
361    
362                            /* If there was an error, or the message is empty, do nothing */
363                            if (isError() || APIUtils.isEmpty(responseText))
364                                    return;
365    
366                            JSONObject responseObject = null;
367    
368                            try {
369                                    responseObject = new JSONObject(responseText);
370    
371                                    if (!responseObject.isNull("Image")) {
372                                            JSONObject imageObject = responseObject
373                                                            .getJSONObject("Image");
374    
375                                            imageID = JSONUtils.getIntegerSafely(imageObject, "id");
376                                            imageKey = JSONUtils.getStringSafely(imageObject, "Key");
377                                    }
378                            } catch (JSONException e) {
379                                    RuntimeJSONException rje = new RuntimeJSONException(e);
380                                    logger.error("An error occured parsing the JSON response", rje);
381                                    throw rje;
382                            }
383                    }
384    
385                    @Override
386                    public String toString() {
387                            return UploadResponse.class.getName() + "[isError=" + isError()
388                                            + ", imageID=" + getImageID() + ", imageKey="
389                                            + getImageKey() + "]";
390                    }
391    
392                    /**
393                     * Used to get the ID of the uploaded image.
394                     * 
395                     * @return the ID of the uploaded image.
396                     */
397                    public Integer getImageID() {
398                            return imageID;
399                    }
400    
401                    /**
402                     * Used to get the security key of the uploaded image.
403                     * 
404                     * @return the security key of the uploaded image.
405                     */
406                    public String getImageKey() {
407                            return imageKey;
408                    }
409            }
