015    package com.kallasoft.smugmug.api.json.v1_2_0.login;
016    
017    import org.json.JSONException;
018    import org.json.JSONObject;
019    import org.slf4j.Logger;
020    import org.slf4j.LoggerFactory;
021    
022    import com.kallasoft.smugmug.api.json.AbstractMethod;
023    import com.kallasoft.smugmug.api.json.AbstractResponse;
024    import com.kallasoft.smugmug.api.json.RuntimeJSONException;
025    import com.kallasoft.smugmug.api.json.util.JSONUtils;
026    import com.kallasoft.smugmug.api.util.APIUtils;
027    
028    /**
029     * This method establishes a session and logs a user in based on the specified
030     * email address and password.
031     * <p>
032     * NOTE: This is a plain-text password, and thus, should only be sent via the
033     * secure server URL.
034     * <p>
035     * Most of the other methods require a SessionID, which is returned by this
036     * method. Store temporarily and re-use it for all other methods which require a
037     * SessionID.
038     * 
039     * @author Riyad Kalla
040     * @version 1.2.0
041     * @see <a
042     *      href="http://smugmug.jot.com/WikiHome/1.2.0/smugmug.login.withPassword">smugmug.login.withPassword
043     *      API Doc</a>
044     */
045    public class WithPassword extends AbstractMethod {
046            /**
047             * Defines the SmugMug JSON API method name that will be called.
048             */
049            public static final String METHOD_NAME = "smugmug.login.withPassword";
050    
051            /**
052             * Defines all the arguments this method takes.
053             * <p>
054             * Values are: "APIKey", "EmailAddress", "Password"
055             */
056            public static final String[] ARGUMENTS = { "APIKey", "EmailAddress",
057                            "Password" };
058    
059            private static final Logger logger = LoggerFactory
060                            .getLogger(WithPassword.class);
061    
062            /**
063             * Construct a new method instance that can be executed.
064             */
065            public WithPassword() {
066                    this(METHOD_NAME, ARGUMENTS);
067            }
068    
069            /**
070             * Construct a new method instance that can be executed with the given
071             * arguments.
072             * 
073             * @param methodName
074             *            The name of the SmugMug JSON API method that this
075             *            <em>Method</em> represents.
076             * @param arguments
077             *            The names of the arguments that this method accepts.
078             */
079            public WithPassword(String methodName, String[] arguments) {
080                    super(methodName, arguments);
081            }
082    
083            /**
084             * Used to execute the smugmug.login.withPassword method, logging the user
085             * in and returning a SessionID that can be used for all other API methods.
086             * 
087             * @param url
088             *            The URL of the SmugMug server to communicate with. Callers are
089             *            encouraged to always use the secure server URL but are not
090             *            restricted from using an alternative URL.
091             * @param argumentValues
092             *            The argument values to pass to this method.
093             * 
094             * @return the response that includes all the information about the logged
095             *         in user.
096             */
097            public WithPasswordResponse execute(String url, String[] argumentValues) {
098                    return new WithPasswordResponse(executeImpl(url, argumentValues));
099            }
100    
101            /**
102             * Convenience method used to execute the smugmug.login.withPassword method.
103             * <p>
104             * This method performs necessary conversions on all the argument values
105             * before calling {@link #execute(String, String[])}.
106             * 
107             * @param url
108             *            The URL of the SmugMug server to communicate with. Callers are
109             *            encouraged to always use the secure server URL but are not
110             *            restricted from using an alternative URL.
111             * @param apiKey
112             *            The API Key to use. API keys are issued by SmugMug.
113             * @param emailAddress
114             *            The email address (login) of the user logging in.
115             * @param password
116             *            The password of the user logging in.
117             * 
118             * @return the response that includes all the information about the logged
119             *         in user.
120             * 
121             * @see #execute(String, String[])
122             */
123            public WithPasswordResponse execute(String url, String apiKey,
124                            String emailAddress, String password) {
125                    return execute(url, new String[] { apiKey, emailAddress, password });
126            }
127    
128            /**
129             * Class used to represent the response for the smugmug.login.withPassword
130             * method call.
131             * 
132             * @author Riyad Kalla
133             * @version 1.2.0
134             */
135            public class WithPasswordResponse extends AbstractResponse {
136                    private Integer userID;
137    
138                    private String sessionID;
139    
140                    private String passwordHash;
141    
142                    private String nickName;
143    
144                    private String displayName;
145    
146                    private String accountType;
147    
148                    private Integer fileSizeLimit;
149    
150                    /**
151                     * Construct a response by parsing the necessary values out of the JSON
152                     * response text.
153                     * 
154                     * @param responseText
155                     *            The JSON-formatted response text that came back from the
156                     *            SmugMug API call.
157                     * 
158                     * @throws RuntimeJSONException
159                     *             if an error occurs while parsing the JSON response text.
160                     */
161                    public WithPasswordResponse(String responseText)
162                                    throws RuntimeJSONException {
163                            /* Parse base response and any error if necessary */
164                            super(responseText);
165    
166                            /* If there was an error, or the message is empty, do nothing */
167                            if (isError() || APIUtils.isEmpty(responseText))
168                                    return;
169    
170                            JSONObject responseObject = null;
171    
172                            try {
173                                    responseObject = new JSONObject(responseText);
174    
175                                    JSONObject loginObject = responseObject.getJSONObject("Login");
176                                    JSONObject userObject = loginObject.getJSONObject("User");
177    
178                                    userID = JSONUtils.getIntegerSafely(userObject, "id");
179                                    nickName = JSONUtils.getStringSafely(userObject, "NickName");
180                                    displayName = JSONUtils.getStringSafely(userObject,
181                                                    "DisplayName");
182    
183                                    passwordHash = JSONUtils.getStringSafely(loginObject,
184                                                    "PasswordHash");
185                                    accountType = JSONUtils.getStringSafely(loginObject,
186                                                    "AccountType");
187                                    fileSizeLimit = JSONUtils.getIntegerSafely(loginObject,
188                                                    "FileSizeLimit");
189                                    sessionID = loginObject.getJSONObject("Session")
190                                                    .getString("id");
191                            } catch (JSONException e) {
192                                    RuntimeJSONException rje = new RuntimeJSONException(e);
193                                    logger.error("An error occured parsing the JSON response", rje);
194                                    throw rje;
195                            }
196                    }
197    
198                    @Override
199                    public String toString() {
200                            return WithPasswordResponse.class.getName() + "[isError="
201                                            + isError() + ", userID=" + getUserID() + ", sessionID="
202                                            + getSessionID() + ", passwordHash=" + getPasswordHash()
203                                            + ", nickName=" + getNickName() + ", displayName="
204                                            + getDisplayName() + ", accountType=" + getAccountType()
205                                            + ", fileSizeLimit=" + getFileSizeLimit() + "]";
206                    }
207    
208                    /**
209                     * Used to get the ID of the logged in user.
210                     * 
211                     * @return the ID of the logged in user.
212                     */
213                    public Integer getUserID() {
214                            return userID;
215                    }
216    
217                    /**
218                     * Used to get the Session ID assigned to this logged in user by
219                     * SmugMug.
220                     * 
221                     * @return the Session ID assigned to this logged in user by SmugMug.
222                     */
223                    public String getSessionID() {
224                            return sessionID;
225                    }
226    
227                    /**
228                     * Used to get the hash of the user's password.
229                     * <p>
230                     * Using this value can be a slightly more secure method of logging a
231                     * user in using {@link WithHash} since the password itself is never
232                     * sent.
233                     * <p>
234                     * For example if the user wanted to store their password client-side in
235                     * a config file somewhere for an application, an implementor might
236                     * consider storing the password hash itself as the password, then
237                     * logging the user back in from that point forward using
238                     * {@link WithHash} instead of this method.
239                     * 
240                     * @return the hash of the user's password.
241                     */
242                    public String getPasswordHash() {
243                            return passwordHash;
244                    }
245    
246                    /**
247                     * Used to get the nick name of the logged in user.
248                     * 
249                     * @return the nick name of the logged in user.
250                     */
251                    public String getNickName() {
252                            return nickName;
253                    }
254    
255                    /**
256                     * Used to get the display name of the logged in user.
257                     * 
258                     * @return the display name of the logged in user.
259                     */
260                    public String getDisplayName() {
261                            return displayName;
262                    }
263    
264                    /**
265                     * Used to get the account type (e.g. Pro) of the logged in user.
266                     * 
267                     * @return the account type (e.g. Pro) of the logged in user.
268                     */
269                    public String getAccountType() {
270                            return accountType;
271                    }
272    
273                    /**
274                     * Used to get the upload file size limit for the logged in user in
275                     * bytes.
276                     * 
277                     * @return the upload file size limit for the logged in user in bytes.
278                     */
279                    public Integer getFileSizeLimit() {
280                            return fileSizeLimit;
281                    }
282            }
