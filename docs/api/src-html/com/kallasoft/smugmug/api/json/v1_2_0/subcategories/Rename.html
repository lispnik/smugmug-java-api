015    package com.kallasoft.smugmug.api.json.v1_2_0.subcategories;
016    
017    import com.kallasoft.smugmug.api.json.AbstractMethod;
018    import com.kallasoft.smugmug.api.json.AbstractResponse;
019    import com.kallasoft.smugmug.api.json.RuntimeJSONException;
020    import com.kallasoft.smugmug.api.util.APIUtils;
021    
022    /**
023     * This method renames the specified SubCategory with new Name.
024     * 
025     * @author Riyad Kalla
026     * @version 1.2.0
027     * @see <a
028     *      href="http://smugmug.jot.com/WikiHome/1.2.0/smugmug.subcategories.rename">smugmug.subcategories.rename
029     *      API Doc</a>
030     */
031    public class Rename extends AbstractMethod {
032            /**
033             * Defines the SmugMug JSON API method name that will be called.
034             */
035            public static final String METHOD_NAME = "smugmug.subcategories.rename";
036    
037            /**
038             * Defines all the arguments this method takes.
039             * <p>
040             * Values are: "APIKey", "SessionID", "SubCategoryID", "Name"
041             */
042            public static final String[] ARGUMENTS = { "APIKey", "SessionID",
043                            "SubCategoryID", "Name" };
044    
045            /**
046             * Construct a new method instance that can be executed.
047             */
048            public Rename() {
049                    this(METHOD_NAME, ARGUMENTS);
050            }
051    
052            /**
053             * Construct a new method instance that can be executed with the given
054             * arguments.
055             * 
056             * @param methodName
057             *            The name of the SmugMug JSON API method that this
058             *            <em>Method</em> represents.
059             * @param arguments
060             *            The names of the arguments that this method accepts.
061             */
062            public Rename(String methodName, String[] arguments) {
063                    super(methodName, arguments);
064            }
065    
066            /**
067             * Used to execute the smugmug.subcategories.rename method, renaming the
068             * given subcategory with the given name.
069             * 
070             * @param url
071             *            The URL of the SmugMug server to communicate with.
072             * @param argumentValues
073             *            The argument values to pass to this method.
074             * 
075             * @return the response that includes the status of the operation.
076             */
077            public RenameResponse execute(String url, String[] argumentValues) {
078                    return new RenameResponse(executeImpl(url, argumentValues));
079            }
080    
081            /**
082             * Convenience method used to execute the smugmug.subcategories.rename
083             * method, renaming the given subcategory with the given name.
084             * <p>
085             * This method performs necessary conversions on all the argument values
086             * before calling {@link #execute(String, String[])}.
087             * 
088             * @param url
089             *            The URL of the SmugMug server to communicate with.
090             * @param apiKey
091             *            The API Key to use. API keys are issued by SmugMug.
092             * @param sessionID
093             *            The logged in SessionID that represents the user's session.
094             * @param subCategoryID
095             *            The ID of the subcategory that will be deleted.
096             * @param name
097             *            The new name for the subcategory.
098             * 
099             * @return the response that includes the status of the operation.
100             * 
101             * @see #execute(String, String[])
102             */
103            public RenameResponse execute(String url, String apiKey, String sessionID,
104                            Integer subCategoryID, String name) {
105                    return execute(url, new String[] { apiKey, sessionID,
106                                    APIUtils.toString(subCategoryID), name });
107            }
108    
109            /**
110             * Class used to represent the response for the smugmug.categories.rename
111             * method call.
112             * 
113             * @author Riyad Kalla
114             * @version 1.2.0
115             */
116            public class RenameResponse extends AbstractResponse {
117                    /**
118                     * Construct a response by parsing the necessary values out of the JSON
119                     * response text.
120                     * 
121                     * @param responseText
122                     *            The JSON-formatted response text that came back from the
123                     *            SmugMug API call.
124                     * 
125                     * @throws RuntimeJSONException
126                     *             if an error occurs while parsing the JSON response text.
127                     */
128                    public RenameResponse(String responseText) throws RuntimeJSONException {
129                            /* Parse base response and any error if necessary */
130                            super(responseText);
131                    }
132    
133                    @Override
134                    public String toString() {
135                            return RenameResponse.class.getName() + "[isError=" + isError()
136                                            + "]";
137                    }
138            }
139    }
