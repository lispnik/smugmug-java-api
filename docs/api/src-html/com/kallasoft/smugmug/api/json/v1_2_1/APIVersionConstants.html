027    /**
028     * Class used to define constants that are specific to this version of the JSON
029     * API.
030     * 
031     * @author Riyad Kalla
032     * @version 1.2.1
033     */
034    public class APIVersionConstants {
035            /**
036             * Used to define a String representing the 1.2.1 version of the SmugMug
037             * API.
038             */
039            public static final String VERSION = "1.2.1";
040    
041            /**
042             * Used to define the SmugMug API 1.2.1 secure (HTTPS) server URL that can
043             * be used to execute any method. The request and response to and from
044             * SmugMug will take place over a secure channel instead of in plain-text.
045             * <p>
046             * It is <strong>highly recommended</strong> that all login methods only
047             * use this URL when communicating with the SmugMug server.
048             */
049            public static final String SECURE_SERVER_URL = "https://api.smugmug.com/services/api/json/1.2.1/";
050    
051            /**
052             * Used to define the SmugMug API 1.2.1 unsecured (HTTP) server URL that can
053             * be used to execute any method. The request and response to and from
054             * SmugMug will be sent in plain-text.
055             * <p>
056             * It is <strong>highly recommended</strong> that you <strong>do not</strong>
057             * use this URL with any of the login methods, and instead use the
058             * {@link #SECURE_SERVER_URL} if possible.
059             */
060            public static final String UNSECURE_SERVER_URL = "http://api.smugmug.com/services/api/json/1.2.1/";
061    
062            /**
063             * Used to define the SmugMug API 1.2.1 Binary upload server URL (HTTP).
064             * This URL has to be used with Binary-Only upload methods like
065             * {@link UploadHTTPPut}.
066             * <p>
067             * Attempting to upload to any other URL will result in a no-op essentially
068             * as the picture will not be stored anywhere.
069             */
070            public static final String BINARY_UPLOAD_SERVER_URL = "http://upload.smugmug.com/";
071    
072            /**
073             * Used to define the SmugMug API 1.2.1 Textual upload server URL (HTTP).
074             * This URL has to be used with Text-Only upload methods like {@link Upload}
075             * and {@link UploadFromURL}.
076             * <p>
077             * Attempting to upload to any other URL will result in a no-op essentially
078             * as the picture will not be stored anywhere.
079             */
080            public static final String TEXT_UPLOAD_SERVER_URL = "http://upload.smugmug.com/services/api/json/1.2.1/";
081    }
