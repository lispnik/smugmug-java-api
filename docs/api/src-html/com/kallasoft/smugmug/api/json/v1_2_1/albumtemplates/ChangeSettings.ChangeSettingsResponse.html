023    /**
024     * This method will change the settings on an album template specified by
025     * AlbumTemplateID.
026     * <p>
027     * The valid values for <code>SortMethod</code> and <code>TemplateID</code>
028     * can be found defined as constants in the {@link Album} class.
029     * 
030     * @author Riyad Kalla
031     * @version 1.2.1
032     * @see <a
033     *      href="http://smugmug.jot.com/WikiHome/1.2.1/smugmug.albumtemplates.changeSettings">smugmug.albumtemplatess.changeSettings
034     *      API Doc</a>
035     */
036    public class ChangeSettings extends AbstractMethod {
037            /* Parameters all copied from the Album class, which is the master copy */
038    
039            /**
040             * Defines the SmugMug JSON API method name that will be called.
041             */
042            public static final String METHOD_NAME = "smugmug.albumtemplates.changeSettings";
043    
044            /**
045             * Defines all the arguments this method takes.
046             * <p>
047             * Values are: "APIKey", "SessionID", "AlbumTemplateID",
048             * "AlbumTemplateName", "Geography", "EXIF", "Clean", "Header", "Filenames",
049             * "TemplateID", "SortMethod", "SortDirection", "Password", "PasswordHint",
050             * "Protected", "Public", "HideOwner", "External", "SmugSearchable",
051             * "WorldSearchable", "Larges", "XLarges", "X2Larges", "X3Larges",
052             * "Originals", "Watermarking", "WatermarkID", "Share", "CanRank",
053             * "Comments", "FamilyEdit", "FriendEdit", "CommunityID", "Printable",
054             * "ProofDays", "Backprinting", "DefaultColor", "UnsharpAmount",
055             * "UnsharpRadius", "UnsharpThreshold", "UnsharpSigma"
056             */
057            public static final String[] ARGUMENTS = { "APIKey", "SessionID",
058                            "AlbumTemplateID", "AlbumTemplateName", "Geography", "EXIF",
059                            "Clean", "Header", "Filenames", "TemplateID", "SortMethod",
060                            "SortDirection", "Password", "PasswordHint", "Protected", "Public",
061                            "HideOwner", "External", "SmugSearchable", "WorldSearchable",
062                            "Larges", "XLarges", "X2Larges", "X3Larges", "Originals",
063                            "Watermarking", "WatermarkID", "Share", "CanRank", "Comments",
064                            "FamilyEdit", "FriendEdit", "CommunityID", "Printable",
065                            "ProofDays", "Backprinting", "DefaultColor", "UnsharpAmount",
066                            "UnsharpRadius", "UnsharpThreshold", "UnsharpSigma" };
067    
068            /**
069             * Construct a new method instance that can be executed.
070             */
071            public ChangeSettings() {
072                    this(METHOD_NAME, ARGUMENTS);
073            }
074    
075            /**
076             * Construct a new method instance that can be executed with the given
077             * arguments.
078             * 
079             * @param methodName
080             *            The name of the SmugMug JSON API method that this
081             *            <em>Method</em> represents.
082             * @param arguments
083             *            The names of the arguments that this method accepts.
084             */
085            public ChangeSettings(String methodName, String[] arguments) {
086                    super(methodName, arguments);
087            }
088    
089            /**
090             * Used to execute the smugmug.albumtemplates.changeSettings method,
091             * changing any settings specified and leaving any settings not specified as
092             * they are.
093             * 
094             * @param url
095             *            The URL of the SmugMug server to communicate with.
096             * @param argumentValues
097             *            The argument values to pass to this method.
098             * 
099             * @return the response returned after the settings have changed.
100             */
101            public ChangeSettingsResponse execute(String url, String[] argumentValues) {
102                    return new ChangeSettingsResponse(executeImpl(url, argumentValues));
103            }
104    
105            /**
106             * Convenience method used to execute the
107             * smugmug.albumtemplates.changeSettings method.
108             * <p>
109             * This method performs necessary conversions on all the argument values
110             * before calling {@link #execute(String, String[])}.
111             * 
112             * @param url
113             *            The URL of the SmugMug server to communicate with.
114             * @param apiKey
115             *            The API Key to use. API keys are issued by SmugMug.
116             * @param sessionID
117             *            The logged in SessionID that represents the user's session.
118             * @param albumTemplateID
119             *            The ID of the album template to apply the changes to.
120             * @param albumTemplateName
121             *            The name of the album template.
122             * @param geography
123             *            <code>true</code> if the album allows the use of geo-tagging
124             *            images (location), otherwise <code>false</code>.
125             * @param exif
126             *            <code>true</code> if the album allows accessing the EXIF
127             *            image data, otherwise <code>false</code>.
128             * @param clean
129             *            <code>true</code> if the album should display using the
130             *            "clean" style, otherwise <code>false</code>.
131             * @param header
132             *            <code>true</code> if the album will display a custom header,
133             *            otherwise <code>false</code>.
134             * @param filenames
135             *            <code>true</code> if the album allows filenames to be shown
136             *            if there is no caption, otherwise <code>false</code>.
137             * @param templateID
138             *            The ID of the template that will be used to display the album.
139             *            <p>
140             *            This can be:
141             *            <ul>
142             *            <li>Viewer Choice (Recommended!): 0</li>
143             *            <li>SmugMug: 3</li>
144             *            <li>Traditional: 4</li>
145             *            <li>All Thumbs: 7</li>
146             *            <li>Slideshow: 8</li>
147             *            <li>Journal: 9</li>
148             *            <li>SmugMug Small: 10</li>
149             *            <li>Filmstrip: 11</li>
150             *            </ul>
151             * @param sortMethod
152             *            The sort method used for the images in this album.
153             *            <p>
154             *            This can be:
155             *            <ul>
156             *            <li>"Position" (Sorts by user-specified position)</li>
157             *            <li>"Caption" (Sorts by the image captions)</li>
158             *            <li>"FileName" (Sorts by the filename of each photo)</li>
159             *            <li>"Date" (Sorts by the date uploaded to SmugMug)</li>
160             *            <li>"DateTime" (Sorts by the date last modified, as told by
161             *            EXIF data - Many files don't have this field set correctly)</li>
162             *            <li>"DateTimeOriginal" (Sorts by the date taken, as told by
163             *            EXIF data - Many cameras don't report this properly)</li>
164             *            </ul>
165             * @param sortDirection
166             *            <code>false</code> (0) for ascending and <code>true</code>
167             *            (1) for descending.
168             * @param position
169             *            The position of the album within it's category (starting with
170             *            1).
171             * @param password
172             *            The password used to password-protect the album if it is
173             *            password protected.
174             * @param passwordHint
175             *            The password hint for the password if there is one.
176             * @param isProtected
177             *            <code>true</code> if the album is using heavy image
178             *            protection, otherwise <code>false</code>.
179             * @param isPublic
180             *            <code>true</code> if the album is publicly accessible,
181             *            otherwise <code>false</code>.
182             * @param hideOwner
183             *            <code>true</code> if the album should mask the URL and name
184             *            of the image owner, otherwise <code>false</code>.
185             * @param external
186             *            <code>true</code> if the album allows external linking,
187             *            otherwise <code>false</code>.
188             * @param smugSearchable
189             *            <code>true</code> if the album is SmugMug-searchable,
190             *            otherwise <code>false</code>.
191             * @param worldSearchable
192             *            <code>true</code> if the album is world-searchable (e.g.
193             *            Google, Yahoo, etc.), otherwise <code>false</code>.
194             * @param larges
195             *            <code>true</code> if the album allows viewing of large-size
196             *            images, otherwise <code>false</code>.
197             * @param xLarges
198             *            <code>true</code> if the album allows viewing of xlarge-size
199             *            images, otherwise <code>false</code>.
200             * @param x2Larges
201             *            <code>true</code> if the album allows viewing of
202             *            xxlarge-size images, otherwise <code>false</code>.
203             * @param x3Larges
204             *            <code>true</code> if the album allows viewing of
205             *            xxxlarge-size images, otherwise <code>false</code>.
206             * @param originals
207             *            <code>true</code> if the album allows the image's originals
208             *            to be seen, otherwise <code>false</code>.
209             * @param watermarking
210             *            <code>true</code> if the album uses watermarking, otherwise
211             *            <code>false</code>.
212             * @param watermarkID
213             *            The ID of the watermark that will be used on the images.
214             * @param share
215             *            <code>true</code> if the album enables the "Easy Sharing"
216             *            button and features, otherwise <code>false</code>.
217             * @param canRank
218             *            <code>true</code> if the album can be ranked, otherwise
219             *            <code>false</code>.
220             * @param comments
221             *            <code>true</code> if the album allows comments, otherwise
222             *            <code>false</code>.
223             * @param familyEdit
224             *            <code>true</code> if the album allows captions and keywords
225             *            to be edited by "family", otherwise <code>false</code>.
226             * @param friendEdit
227             *            <code>true</code> if the album allows captions and keywords
228             *            to be edited by "friends", otherwise <code>false</code>.
229             * @param communityID
230             *            The ID of the community that this album belongs to.
231             * @param printable
232             *            <code>true</code> if the album allows prints to be
233             *            purchased, otherwise <code>false</code>.
234             * @param proofDays
235             *            The number of days to delay printing for this album.
236             * @param backprinting
237             *            The optional text that can be printed on the backs of the
238             *            prints.
239             * @param defaultColor
240             *            <code>true</code> if the album defaults to using "True" or
241             *            "AutoColor" when it's in the cart, otherwise
242             *            <code>false</code>.
243             * @param unsharpAmount
244             *            How weak or strong the sharpness mask will be. A higher value
245             *            will result in a more stark sharpness (Default 0.200). This
246             *            only applies to resized versions of the original image (Large,
247             *            XLarge, etc.) See <a
248             *            href="http://smugmug.com/help/display-quality">http://smugmug.com/help/display-quality</a>
249             *            for more information.
250             * @param unsharpRadius
251             *            The radius on the sharpening effect. Larger values lead to
252             *            more defined sharpening (Default 1.000). This only applies to
253             *            resized versions of the original image (Large, XLarge, etc.)
254             *            See <a
255             *            href="http://smugmug.com/help/display-quality">http://smugmug.com/help/display-quality</a>
256             *            for more information.
257             * @param unsharpThreshold
258             *            The number of adjacent pixels that are considered an "edge"
259             *            before performing a sharpen (Default 0.050). This only applies
260             *            to resized versions of the original image (Large, XLarge,
261             *            etc.) See <a
262             *            href="http://smugmug.com/help/display-quality">http://smugmug.com/help/display-quality</a>
263             *            for more information.
264             * @param unsharpSigma
265             *            (Default 1.000)
266             * 
267             * @return the response returned after the settings have changed.
268             * 
269             * @see #execute(String, String[])
270             */
271            public ChangeSettingsResponse execute(String url, String apiKey,
272                            String sessionID, Integer albumTemplateID,
273                            String albumTemplateName, Boolean geography, Boolean exif,
274                            Boolean clean, Boolean header, Boolean filenames,
275                            Integer templateID, String sortMethod, Boolean sortDirection,
276                            Integer position, String password, String passwordHint,
277                            Boolean isProtected, Boolean isPublic, Boolean hideOwner,
278                            Boolean external, Boolean smugSearchable, Boolean worldSearchable,
279                            Boolean larges, Boolean xLarges, Boolean x2Larges,
280                            Boolean x3Larges, Boolean originals, Boolean watermarking,
281                            Integer watermarkID, Boolean share, Boolean canRank,
282                            Boolean comments, Boolean familyEdit, Boolean friendEdit,
283                            Integer communityID, Boolean printable, Integer proofDays,
284                            String backprinting, Boolean defaultColor, Float unsharpAmount,
285                            Float unsharpRadius, Float unsharpThreshold, Float unsharpSigma) {
286                    return execute(url, new String[] { apiKey, sessionID,
287                                    APIUtils.toString(albumTemplateID), albumTemplateName,
288                                    APIUtils.toString(geography), APIUtils.toString(exif),
289                                    APIUtils.toString(clean), APIUtils.toString(header),
290                                    APIUtils.toString(filenames), APIUtils.toString(templateID),
291                                    sortMethod, APIUtils.toString(sortDirection),
292                                    APIUtils.toString(position), password, passwordHint,
293                                    APIUtils.toString(isProtected), APIUtils.toString(isPublic),
294                                    APIUtils.toString(hideOwner), APIUtils.toString(external),
295                                    APIUtils.toString(smugSearchable),
296                                    APIUtils.toString(worldSearchable), APIUtils.toString(larges),
297                                    APIUtils.toString(xLarges), APIUtils.toString(x2Larges),
298                                    APIUtils.toString(x3Larges), APIUtils.toString(originals),
299                                    APIUtils.toString(watermarking),
300                                    APIUtils.toString(watermarkID), APIUtils.toString(share),
301                                    APIUtils.toString(canRank), APIUtils.toString(comments),
302                                    APIUtils.toString(familyEdit), APIUtils.toString(friendEdit),
303                                    APIUtils.toString(communityID), APIUtils.toString(printable),
304                                    APIUtils.toString(proofDays), APIUtils.toString(backprinting),
305                                    APIUtils.toString(defaultColor),
306                                    APIUtils.toString(unsharpAmount),
307                                    APIUtils.toString(unsharpRadius),
308                                    APIUtils.toString(unsharpThreshold),
309                                    APIUtils.toString(unsharpSigma) });
310            }

312            /**
313             * Class used to represent the response for the
314             * smugmug.albumtemplates.changeSettings method call.
315             * <p>
316             * This response doesn't return any additional arguments. The stat code
317             * returned by {@link AbstractResponse#getStat()} indicates if the operation
318             * was successful or not.
319             * 
320             * @author Riyad Kalla
321             * @version 1.2.1
322             */
323            public class ChangeSettingsResponse extends AbstractResponse {
324                    /**
325                     * Construct a response by parsing the necessary values out of the JSON
326                     * response text.
327                     * 
328                     * @param responseText
329                     *            The JSON-formatted response text that came back from the
330                     *            SmugMug API call.
331                     * 
332                     * @throws RuntimeJSONException
333                     *             if an error occurs while parsing the JSON response text.
334                     */
335                    public ChangeSettingsResponse(String responseText)
336                                    throws RuntimeJSONException {
337                            super(responseText);
338                    }
339    
340                    @Override
341                    public String toString() {
342                            return ChangeSettingsResponse.class.getName() + "[isError="
343                                            + isError() + "]";
344                    }
345            }
