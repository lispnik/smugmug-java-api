015    package com.kallasoft.smugmug.api.json.v1_2_1.images;
016    
017    import java.io.InputStream;
018    
019    import org.slf4j.Logger;
020    import org.slf4j.LoggerFactory;
021    
022    import com.kallasoft.smugmug.api.json.v1_2_1.APIVersionConstants;
023    
024    /**
025     * This class is a convenience class extending the base
026     * {@link com.kallasoft.smugmug.api.json.v1_2_0.images.Upload} implementation.
027     * 
028     * @author Riyad Kalla
029     * @version 1.2.1
030     * @see com.kallasoft.smugmug.api.json.v1_2_0.images.Upload
031     */
032    public class Upload extends com.kallasoft.smugmug.api.json.v1_2_0.images.Upload {
033            private static final Logger logger = LoggerFactory.getLogger(Upload.class);
034    
035            /**
036             * Construct a new method instance that can be executed.
037             */
038            public Upload() {
039                    this(METHOD_NAME, ARGUMENTS);
040            }
041    
042            /**
043             * Construct a new method instance that can be executed with the given
044             * arguments.
045             * 
046             * @param methodName
047             *            The name of the SmugMug JSON API method that this
048             *            <em>Method</em> represents.
049             * @param arguments
050             *            The names of the arguments that this method accepts.
051             */
052            public Upload(String methodName, String[] arguments) {
053                    super(methodName, arguments);
054            }
055    
056            @Override
057            public UploadResponse execute(String url, String[] argumentValues) {
058                    if (!APIVersionConstants.TEXT_UPLOAD_SERVER_URL.equals(url))
059                            logger
060                                            .warn(
061                                                            "url [{}] should normally be equal to the Text-based SmugMug Upload Server URL (defined by APIVersionConstants.TEXT_UPLOAD_SERVER_URL), otherwise the uploaded image won't be processed.",
062                                                            url);
063    
064                    /*
065                     * This method has to assume that the String[] being passed in has been
066                     * setup exactly as the caller wants it to be, so no additional
067                     * pre-processing (e.g. calling one of the methods that calculates MD5,
068                     * encoding, etc.) should be performed on the contents of this array.
069                     * The arguments should be passed through immediately, as-is, to
070                     * super.executeImpl.
071                     */
072                    return new UploadResponse(executeImpl(url, argumentValues));
073            }
074    
075            @Override
076            public UploadResponse execute(String url, String apiKey, String sessionID,
077                            Integer albumID, String fileName, InputStream inputStream) {
078                    return execute(url, apiKey, sessionID, albumID, fileName, inputStream,
079                                    null, null, null, null, null);
080            }
081    
082            @Override
083            public UploadResponse execute(String url, String apiKey, String sessionID,
084                            Integer albumID, String fileName, InputStream inputStream,
085                            String caption, String keywords, Double latitude, Double longitude,
086                            Double altitude) throws IllegalArgumentException, RuntimeException {
087                    return execute(url, prepareUploadArgumentValues(apiKey,
088                                    sessionID, albumID, fileName, inputStream, caption, keywords,
089                                    latitude, longitude, altitude));
090            }
091    
092            /**
093             * This class is a convenience class extending the base
094             * {@link com.kallasoft.smugmug.api.json.v1_2_0.images.Upload.UploadResponse}
095             * implementation.
096             * 
097             * @author Riyad Kalla
098             * @version 1.2.1
099             * @see com.kallasoft.smugmug.api.json.v1_2_0.images.Upload.UploadResponse
100             */
101            public class UploadResponse extends
102                            com.kallasoft.smugmug.api.json.v1_2_0.images.Upload.UploadResponse {
103                    public UploadResponse(String responseText) {
104                            super(responseText);
105                    }
106            }
107    }
